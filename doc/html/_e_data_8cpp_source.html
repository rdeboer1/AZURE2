287int EData::ReadTargetEffectsFile(const Config& configure, CNuc *compound) {
288 std::ifstream in(configure.configfile.c_str());
289 if(!in) return -1;
290 std::string line="";
291 while(line!="<targetInt>"&&!in.eof()) getline(in,line);
292 if(line!="<targetInt>") return -1;
293 line="";
294 while(line!="</targetInt>"&&!in.eof()) {
295 getline(in,line);
296 bool empty=true;
297 for(unsigned int i=0;i<line.size();++i)
298 if(line[i]!=' '&&line[i]!='\t') {
299 empty=false;
300 break;
301 }
302 if(empty==true) continue;
303 if(line!="</targetInt>"&&!in.eof()){
304 std::istringstream stm;
305 stm.str(line);
306 TargetEffect targetEffect(stm,configure);
307 if(stm.rdstate() & (std::stringstream::failbit | std::stringstream::badbit)) return -1;
308 if(targetEffect.IsActive()) {
309 this->AddTargetEffect(targetEffect);
310 TargetEffect *thisTargetEffect=this->GetTargetEffect(this->NumTargetEffects());
311 std::vector<int> segmentsList = thisTargetEffect->GetSegmentsList();
312 for(int i = 1;i<=segmentsList.size();i++) {
313 if(this->IsSegmentKey(segmentsList[i-1]))
314 this->GetSegmentFromKey(segmentsList[i-1])->SetTargetEffectNum(this->NumTargetEffects());
315 }
316 }
317 }
318 }
319 if(line!="</targetInt>") return -1;
320 for(ESegmentIterator segment=GetSegments().begin();segment<GetSegments().end();segment++) {
321 PPair *entrancePair = compound->GetPair(compound->GetPairNumFromKey(segment->GetEntranceKey()));
322 PPair *exitPair = compound->GetPair(compound->GetPairNumFromKey(segment->GetExitKey()));
323 double cmConversion;
324 if(entrancePair->GetPType()==20)
325 cmConversion = (exitPair->GetM(1)+exitPair->GetM(2))/exitPair->GetM(2);
326 else
327 cmConversion = entrancePair->GetM(2)/(entrancePair->GetM(1)+entrancePair->GetM(2));
328 if(segment->IsTargetEffect()) {
329 TargetEffect *targetEffect = this->GetTargetEffect(segment->GetTargetEffectNum());
330 double sigma = targetEffect->GetSigma();
331 targetEffect->SetSigma(cmConversion*sigma);
332 for(EPointIterator point=segment->GetPoints().begin();point<segment->GetPoints().end();point++) {
333 point->SetTargetEffectNum(segment->GetTargetEffectNum());
334 if(targetEffect->IsTargetIntegration()||targetEffect->IsConvolution()) {
335 double forwardDepth=0.0;
336 double backwardDepth=0.0;
337 if(targetEffect->IsTargetIntegration()) {
338 double totalM=entrancePair->GetM(1)+entrancePair->GetM(2);
339 double targetThickness = cmConversion*targetEffect->TargetThickness(point->GetLabEnergy(),configure);
340 point->SetTargetThickness(targetThickness);
341 if(targetEffect->IsConvolution()) {
342 backwardDepth=targetThickness+targetEffect->convolutionRange*targetEffect->GetSigma();
343 forwardDepth=targetEffect->convolutionRange*targetEffect->GetSigma();
344 } else {
345 backwardDepth=targetThickness;
346 forwardDepth=0.0;
347 }
348 } else if(targetEffect->IsConvolution()) {
349 backwardDepth=targetEffect->convolutionRange*targetEffect->GetSigma();
350 forwardDepth=targetEffect->convolutionRange*targetEffect->GetSigma();
351 }
352 for(int i=0;i<targetEffect->NumSubPoints();i++) {
353 double subEnergy=point->GetCMEnergy()+forwardDepth
354 -(forwardDepth+backwardDepth)/(targetEffect->NumSubPoints())*i;
355 EPoint subPoint(point->GetCMAngle(),subEnergy,&*segment);
356 if(targetEffect->IsTargetIntegration()) {
357 double stoppingPower=cmConversion*targetEffect->
358 GetStoppingPowerEq()->Evaluate(configure,subEnergy/cmConversion);
359 subPoint.SetStoppingPower(stoppingPower);
360 }
361 point->AddSubPoint(subPoint);
362 }
363 }
364 }
365 }
366 }
367 return 0;

447int EData::Initialize(CNuc *compound,const Config &configure) {
448 //Calculate channel lo-matrix and channel penetrability for each channel at each local energy
449 configure.outStream << "Calculating Lo-Matrix, Phases, and Penetrabilities..." << std::endl;
450 if(this->CalcEDependentValues(compound,configure)==-1) return -1;
451 if((configure.fileCheckMask|configure.screenCheckMask) & Config::CHECK_ENERGY_DEP)
452 this->PrintEDependentValues(configure,compound);
453
454 //Calculate legendre polynomials for each data point
455 configure.outStream << "Calculating Legendre Polynomials..." << std::endl;
456 this->CalcLegendreP(configure.maxLOrder);
457 if((configure.fileCheckMask|configure.screenCheckMask) & Config::CHECK_LEGENDRE)
458 this->PrintLegendreP(configure);
459
460 //Calculate Coulomb Amplitudes
461 configure.outStream << "Calculating Coulomb Amplitudes..." << std::endl;
462 this->CalcCoulombAmplitude(compound);
463 if((configure.fileCheckMask|configure.screenCheckMask) & Config::CHECK_COUL_AMPLITUDES) {
464 this->PrintCoulombAmplitude(configure,compound);
465 }
466
467 //Calculate new ec amplitudes
468 if(configure.paramMask & Config::USE_EXTERNAL_CAPTURE) {
469 configure.outStream << "Calculating External Capture Amplitudes..." << std::endl;
470 if(this->CalculateECAmplitudes(compound,configure)==-1) return -1;
471 }
472 return 0;
473}

870int EData::CalculateECAmplitudes(CNuc *theCNuc,const Config& configure) {
871 std::ifstream in;
872 std::ofstream out;
873 std::string outputfile;
874 if(configure.paramMask & Config::CALCULATE_WITH_DATA) outputfile=configure.outputdir+"intEC.dat";
875 else outputfile=configure.outputdir+"intEC.extrap";
876 if(configure.paramMask & Config::USE_PREVIOUS_INTEGRALS) in.open(configure.integralsfile.c_str());
877 else {
878 out.open(outputfile.c_str());
879 if(!out) configure.outStream << "Could not write to EC Amplitude File." << std::endl;
880 }
881 int sumSegmentI=0;
882 int numSumSegments=0;
883 for(ESegmentIterator segment=GetSegments().begin();segment<GetSegments().end();segment++) {
884 if(segment->IsTotalCapture()) {
885 numSumSegments = segment->IsTotalCapture();
886 sumSegmentI=0;
887 }
888 if(numSumSegments) sumSegmentI++;
889 char segmentKeyOut[256];
890 if(numSumSegments) sprintf(segmentKeyOut,"%d (%d/%d)",segment->GetSegmentKey(),sumSegmentI,numSumSegments);
891 else sprintf(segmentKeyOut,"%d",segment->GetSegmentKey());
892 int aa=theCNuc->GetPairNumFromKey(segment->GetEntranceKey());
893 if(theCNuc->GetPair(aa)->GetPType()==20) continue;
894 if(theCNuc->GetPair(aa)->IsEntrance()) {
895 PPair *entrancePair=theCNuc->GetPair(aa);
896 for(int j=1;j<=theCNuc->NumJGroups();j++) {
897 for(int la=1;la<=theCNuc->GetJGroup(j)->NumLevels();la++) {
898 if(theCNuc->GetJGroup(j)->GetLevel(la)->IsECLevel()) {
899 ALevel *ecLevel = theCNuc->GetJGroup(j)->GetLevel(la);
900 int ir=theCNuc->GetPairNumFromKey(segment->GetExitKey());
901 if(ecLevel->GetECPairNum()==ir) {
902 if(!(configure.paramMask & Config::USE_PREVIOUS_INTEGRALS)) {
903 configure.outStream << "\tSegment #" << std::setw(12) << segmentKeyOut
904 << std::setw(0) << " [ ] 0%";configure.outStream.flush();
905 int numPoints=segment->NumPoints();
906 int pointIndex=0;
907 time_t startTime = time(NULL);
908 bool localStop = false;
909#pragma omp parallel for shared(configure,localStop)
910 for(int i=1;i<=numPoints;i++) {
911 if(configure.stopFlag||localStop) continue;
912 EPoint *point = segment->GetPoint(i);
913 if(!(point->IsMapped())) {
914 try {
915 point->CalculateECAmplitudes(theCNuc,configure);
916 } catch(GSLException e) {
917#pragma omp critical
918 {
919 configure.outStream << e.what() << std::endl;
920 localStop=true;
921 }
922 }
923 }
924 ++pointIndex;
925 if(difftime(time(NULL),startTime)>0.25) {
926 startTime=time(NULL);
927 std::string progress=" [";
928 double percent=0.;
929 for(int j = 1;j<=25;j++) {
930 if(pointIndex>=percent*numPoints&&percent<1.) {
931 percent+=0.04;
932 progress+='*';
933 } else progress+=' ';
934 } progress+="] ";
935 configure.outStream << "\r\tSegment #" << std::setw(12) << segmentKeyOut
936 << std::setw(0) << progress << percent*100 << '%';configure.outStream.flush();
937 }
938 }
939 if(configure.stopFlag||localStop) {
940 if(out.is_open()) out.close();
941 if(in.is_open()) in.close();
942 return -1;
943 }
944 configure.outStream << "\r\tSegment #" << std::setw(12) << segmentKeyOut
945 << std::setw(0) << " [*************************] 100%" << std::endl;
946 }
947 for(EPointIterator point=segment->GetPoints().begin();
948 point<segment->GetPoints().end();point++) {
949 if(!(point->IsMapped())) {
950 for(int k=1;k<=entrancePair->GetDecay(ir)->NumKGroups();k++) {
951 for(int ecm=1;ecm<=entrancePair->GetDecay(ir)->GetKGroup(k)->NumECMGroups();ecm++) {
952 if(!(configure.paramMask & Config::USE_PREVIOUS_INTEGRALS)) {
953 if(out.is_open()) out << point->GetECAmplitude(k,ecm) << std::endl;
954 for(EPointIterator subPoint=point->GetSubPoints().begin();
955 subPoint<point->GetSubPoints().end();subPoint++)
956 if(out.is_open()) out << subPoint->GetECAmplitude(k,ecm) << std::endl;
957 } else {
958 complex ecAmplitude(0.0,0.0);
959 in >> ecAmplitude;
960 point->AddECAmplitude(k,ecm,ecAmplitude);
961 for(EPointIterator subPoint=point->GetSubPoints().begin();
962 subPoint<point->GetSubPoints().end();subPoint++) {
963 ecAmplitude=complex(0.0,0.0);
964 in >> ecAmplitude;
965 subPoint->AddECAmplitude(k,ecm,ecAmplitude);
966 }
967 }
968 for(EPointMapIterator mappedPoint=point->GetMappedPoints().begin();
969 mappedPoint<point->GetMappedPoints().begin();mappedPoint++) {
970 (*mappedPoint)->AddECAmplitude(k,ecm,point->GetECAmplitude(k,ecm));
971 for(int i=1;i<=point->NumSubPoints();i++) {
972 (*mappedPoint)->GetSubPoint(i)->
973 AddECAmplitude(k,ecm,point->GetSubPoint(i)->GetECAmplitude(k,ecm));
974 }
975 }
976 }
977 }
978 }
979 }
980 }
981 }
982 }
983 }
984 }
985 if(sumSegmentI==numSumSegments) {
986 sumSegmentI=0;
987 numSumSegments=0;
988 }
989 }
990 if(out.is_open()) {
991 out.flush();
992 out.close();
993 }
994 if(in.is_open()) in.close();
995 return 0;
996}
