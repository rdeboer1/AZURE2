3// Calculation of h(omega) = H(omega)'/H(omega) with a continued fraction.
4// -----------------------------------------------------------------------
5// One calculates the ratio h = H'/H with the continued fraction of the associated hypergeometric confluent function.
6// One uses Lentz's method.
7// One has : h = [b[0] + a[1]/b[1]+ a[2]/b[2]+ ... a[n]/b[n]+ ...].i.omega/z with :
8// b[0] = z - eta, a[n] = (1 + l + i.omega + n-1).(i.omega.eta - l + n-1), b[n] = 2[z - eta] + i.omega.n .
9//
10// If the number of iterations reaches 100000 and |z| > 0.5, the convergence is too slow. One is probably very close to the imaginary axis.
11// If l=0 and |z| <= 0.5, the direct integration is still done as it is stable.
12// If 1+l+/-i.eta is negative integer, it has to be done otherwise it is too long even if |z| is not exceedingly small.
13// One first considers Re[z] >= 0.
14// One takes the starting point z0 = 2 + i.(Im[z] + 2.sign[Im[z]]) (0.6 + i.sign[Re[z]].0.6.sign[Im[z]] if |z| <= 0.5).
15// Then, one calculates H[omega],H[omega]' at the starting point, and one integrates numerically H[omega] until z.
16// h(omega)(z) is then H[omega](z)'/H[omega](z).
17// If Re[z] < 0, one calculates H[-omega],H[-omega]' with l, eta -> -eta, and z -> -z.
18// One uses the Coulomb wave functions class cwf_minus_eta_ptr defined with l and -eta.
19// The ratio h(omega,l,eta,z) is then equal to -h(-omega,l,-eta,-z).
20// To avoid infinite loops, continued_fraction_h must not be used in this integration. So, the starting point is chosen so |H[+/-]| is very likely
21// to increase in modulus. Then, one always integrates forward. Forward integration is enforced putting
22// is_H_dir_int_naive to true. It is put to false again at the end of the calculation.
23//
24// Variables:
25// ----------
26// z : variable of the Coulomb wave function.
27// omega : 1 for the outgoing wave function ratio H+'/H+, -1 for the incoming wave function ratio H-'/H-.
28// large,small : 1E50,1E-50. They are used in the case of vanishing denominators or numerators.
29// I_omega,I_omega_eta,two_I_omega,z_minus_eta,two_z_minus_eta : i.omega,i.omega.eta, 2.i.omega, z-eta, 2(z-eta).
30// a,c : 1 + l + i.omega.eta, i.omega.eta - l.
31// b0,Cn,Dn,an,bn,Delta_n : variables used in the Lentz method.
32// n,nm1 : index of a[n] and b[n], n-1
33// bn_plus_an_Dn : bn + an.Dn. Dn = 1/[bn + an.Dn] or 1E50 if bn + an.Dn is zero.
34// bn_plus_an_over_Cn : bn + an/Cn. Cn = bn + an/Dn or 1E-50 if bn + an/Dn is zero.
35// hn : value of the continuous fraction during the iteration process.
36// test : test of convergence of hn.
37// cwf : reference to *this if Re[z] >= 0, reference to cwf_minus_eta_ptr if Re[z] < 0
38// cwf_minus_eta_ptr is allocated first if it is zero, in the case of Re[z] < 0 .
39// It is used to integrate the Coulomb wave functions with l,eta for Re[z] >= 0 or l,-eta for Re[z] < 0.
40// z00 : 2 + i.sign[Re[z]].(Im[z] + 2.sign[Im[z]]).
41// z01 : 0.6 + i.0.6.sign[Re[z]].sign[Im[z]].
42// abs_z : |z|
43// z_start,F_start,dF_start : starting point of the direct integration, F(l,+/-eta,z_start),F'(l,+/-eta,z_start)
44// H,dH : Coulomb wave function and derivative in l,eta,omega,z if Re[z] > 0, in l,-eta,-omega,-z if Re[z] < 0.
45// h : value of H(omega)'/H(omega).
46// debut_cwf,F_debut_cwf,dF_debut_cwf : values stored in cwf put back in cwf at the end of direct integration as they change values.

99// ------------------------------------------------------------------------
100// H[omega](z) = exp[i.omega.[z - eta.log[2z] - l.Pi/2 + sigma(l,eta)]].S[omega](z) for Re[z] >= 0.
101// H[omega]'(z) = exp[i.omega.[z - eta.log[2z] - l.Pi/2 + sigma(l,eta)]].[S[omega]'(z) + i.omega.(1 - eta/z).S[omega](z)] for Re[z] >= 0.
102//
103// S[omega](z) is the asymptotic series and S[omega]'(z) its derivative calculated in asymptotic_series.
104// If they did not converge, one leaves the routine.
105//
106// The negative cut is taken into account if log [cut_constant_AS] is finite, i.e. cut_constant_AS is not exactly zero :
107//
108// If Re[z] < 0.0 and omega.Im[z] < 0.0 :
109// --------------------------------------
110// H[omega] = H[omega][ASd] + cut_constant_AS_plus.H[-omega][ASd].
111// H[omega][ASd] is given by directly by the asymptotic series.
112// H[-omega][ASd] is given by the asymptotic series, which gives the good result H[-omega] in this case as one is not in its bad quadrant.
113//
114// The function is scaled, so one returns : H[omega](z).exp[-i.omega.[z - eta.log[2z]]] and H[omega]'(z).exp[-i.omega.[z - eta.log[2z]]] .
115//
116// In the case of overflows or underflows, one uses logs.
117//
118//
119// Variables :
120// -----------
121// omega : 1 if one calculates H+(z) and H+'(z), -1 if one calculates H-(z) and H-'(z).
122// one_over_z : 1/z.
123// z : variable of the Coulomb wave function.
124// H_scaled,dH_scaled : H[omega](z).exp(-i.omega.[z - eta.log[2z]]) and H[omega]'(z).exp(-iomega.[z - eta.log[2z]]).
125// is_it_successful : true if the asymptotic expansions converged, false it not
126// sum,dsum : {S[omega](z),S[-omega](z)} and {S[omega]'(z),S[-omega]'(z)}
127// I_omega,two_I_omega : i.omega, 2i.omega.
128// I_omega_one_minus_eta_over_z : i.omega.(1 - eta/z).
129// phase_shift : i.omega.[-l.Pi/2 + sigma(l,eta)]
130// exp_phase_shift : exp (i.omega.(-l.Pi/2 + sigma(l,eta))).
131// log_cut_constant_AS : log_cut_constant_AS_plus if omega = 1, log_cut_constant_AS_minus if omega =-1.
132// If log_cut_constant_AS is not finite (i.e. cut_constant_AS = 0 if it is defined), there is no branch cut to consider.
133// factor : exp (log_cut_constant_AS - 2.i.omega.(z - eta.log(2z)) - phase_shift).

179// Calculation of H[omega](z) and H[omega]'(z) by direct integration.
180// ------------------------------------------------------------------
181// To calculate H[omega](z) and H'[omega](z), one integrates numerically H[omega]''(z) = [l(l+1)/z^2 + 2.eta/z - 1].H[omega](z)
182// starting from debut,H_debut = H[omega](debut) and dH_debut = H'[omega](debut).
183// There is no branch cut problem as Re[z] >= 0.
184// The starting point comes for the regular function from the stored values debut, F_debut and dF_debut.
185// If debut = 0, one puts debut = debut_omega = z/|2z| and calculates F(debut) and F'(debut) with power series.
186// Then, the starting point {debut_omega,H[omega](debut),H'[omega](debut)} is calculated
187// from {debut,F(debut),F'(debut)} and the continued fraction h[omega](debut).
188// The first order expansions method is used if one is very close to the real axes of l,eta and z (Re[z] > 0).
189// The step of the integration is (z - debut)/N_num, with N_num = [|z - debut|/min (0.1,10.turning_point)] + 1.
190// The value of min (0.1,10/turning_point) gives a smaller step when turning_point increases,
191// as calculations become there more difficult as |H[omega]| typically varies faster in this case.
192// The intermediates points are called z_aft. They go from debut to z,
193// and (debut_omega,H_debut,dH_debut) is put to {z_aft,H[omega](z_aft),H'[omega](z_aft)} at each step.
194// If |H[omega]| increases along the path, the integration is stable.
195// If is_H_dir_int_naive is true, one has to integrate forward, as this integration is used to calculate the continued fraction.
196// If not, and if |H[omega]| decreases,
197// one integrates backwards from z_aft to debut_omega with the knowledge of h[omega](z_aft) = H'[omega](z_aft)/H[omega](z_aft).
198// H'[omega](z_aft)/H[omega](z_aft) is given by the continued fraction formula.
199// One then obtains by direct integration C.H[omega](debut) and C.H'[omega](debut).
200// Knowing H_debut, one deduces H[omega](z) = 1/C and H'[omega](z) = f(z_aft).H[omega](z).
201// Increase or decrease is known using the Taylor expansion of H[omega] near debut_omega in z up to second order.
202//
203// If H(z_aft) is not finite, one stops the integration.
204//
205// Variables
206// ---------
207// omega : 1 for H+,H+', -1 for H-,H-'.
208// z : variable of the Coulomb wave function.
209// H,dH : wave function H[omega] and derivative H'[omega] to calculate.
210// x,y,l_r,l_i,eta_r,eta_i : Re[z], Im[z], Re[l], Im[l], Re[eta], Im[eta].
211// debut_omega, H_debut,dH_debut : starting point or the integration. debut_omega = debut at the beginning.
212// ODE : reference to pointer ODE_ptr, which performs direct integration from one point to another.
213// step_abs : length of the integration step from z to debut. It is min (0.1,10/turning_point)
214// N_num : number of integrations to do from debut to z. It is |z - debut|/step_abs + 1.
215// step_num : std::complex step from debut to z. It is (z - debut)/N_num .
216// ll_plus_one,two_eta : l(l+1), 2.eta. They are used in the Taylor expansion test.
217// z_aft : next point in which H[omega] is calculated by direct integration . It is z - i*step_num, with i from N_num-1 to 0.
218// one_over_debut,log_H_debut_der,d2H_debut_over_H_debut: 1/debut, H'[omega](debut)/H[omega](debut), H[omega]''(debut)/H[omega](debut).
219// They are used in the Taylor expansion test.
220// h : continued fraction h[omega] = H[omega]'/H[omega] in z.
221// H_debut_not_normed,dH_debut_not_normed : unnormed regular wave function and derivative in debut_omega,
222// after integration from the starting point {z_aft, 1.0, H'[omega](z_aft)/H[omega](z_aft)}.
223// Then, H[omega](z_aft) = H_debut/H_debut_not_normed and H'[omega](z_aft) = h[omega].H[omega](z_aft) .

334// Calculation of H(omega) and H(omega)' with F(z), F'(z) and continued fractions.
335// -------------------------------------------------------------------------------
336// If 1+l-i.omega.eta is negative integer, one uses H[omega] = 1/F/(f - h[omega]) as it is the only available solution
337// linearly independent of F.
338// If 1+l+i.omega.eta is negative integer, H[omega] is proportional to F so the values F and F' are arbitrarily chosen for H[omega] and H'[omega].
339//
340// If not, one calculates h[sign(Im[z])](z), and one has f = F'(z)/F(z).
341// One chooses sign(Im[z]), as h converges fastest for z in this region.
342// If |f - h[sign(Im[z])]|oo >= 1, F and H[sign] are numerically linearly independent so the continued fraction is meaningful.
343// Then, h[-sign(Im[z])](z) is not needed and is put to f, the worst value it can have. If not, h[-sign(Im[z])](z) is needed and calculated.
344// Then, h[omega] and h[-omega] take their values from h[sign(Im[z])]| and h[-sign(Im[z])]|.
345//
346// If |f - h[omega]|oo > |f - h[-omega]|oo, one uses the continued fraction h[omega].
347// If Re[z] > 0 or sign(Im[z]) = omega (good quadrants), H[omega] = 1/F/(f - h[omega]) and H'[omega] = h[omega].H[omega].
348// If not, one has to take the branch cut into account, as one is in the bad quadrant of H[omega] :
349// H[omega] = 1/F/(f - h[omega]) - cut_constant.F and H'[omega] = h[omega].H[omega] - cut_constant.F' .
350// cut_constant is cut_constant_CFa_plus if omega = 1, and cut_constant_CFa_minus if omega = -1.
351// If log [cut_constant] is not finite, it means that cut_constant is exactly zero if it is defined, so that there is no branch cut to consider.
352//
353// If |f - h[omega]|oo < |f - h[-omega]|oo, one uses the continued fraction h[-omega],
354// calculating H[omega] and H'[omega] using H[omega] = H[-omega] + constant.F .
355// constant is 2.i.omega.norm if Re[z] > 0 or sign(Im[z]) = -omega (good quadrants), and cut_constant if not.
356// cut_constant is cut_constant_CFb_plus if omega = 1, cut_constant_CFb_minus if omega = -1.
357// norm is 1 if is_it_normalized is true, C(l,eta)^2 if not.
358//
359// If cut_constant underflows or overflows, one uses logs of F,F' and log_cut_constant for the calculation.
360//
361// Variables:
362// ----------
363// omega : 1 if one calculates H+(z) and H+'(z), -1 if one calculates H-(z) and H-'(z).
364// z : variable of the Coulomb wave function.
365// F,dF: regular Coulomb wave function and derivative.
366// H,dH : H+(z) and H+'(z) if omega=1, H-(z) and H-'(z) if omega=-1.
367// x,y : Re[z], Im[z].
368// f,two_I_omega : ratio F'(z)/F(z), 2.i.omega .
369// h_sign,h_minus_sign : continuous fractions h[-SIGN[y]] and h[SIGN[y]]. See before for their calculations
370// h_omega,h_minus_omega : continuous fraction h[omega] and h[-omega].

452// When 2l is non-integer, one can expand H[omega] with F[l,eta,z] and F[-l-1,eta,z].

521// Calculation of F and F' by power series.
522// ----------------------------------------
523// It is used only when |z| <= 0.5, to avoid numerical inaccuracies.
524//
525// F(z) = norm.z^(l+1).\sum a[n], n in [0:+oo[, where :
526// a[0] = 1.0.
527// a[1] = z.eta/(l+1).
528// a[n] = (2.z.eta.a[n-1] - a[n-2].(z^2))/(n.(n+2l+1)),n >= 2.
529//
530// The z = 0 case is treated first. It is defined only for Re[l] > 0 or l = 0. The program aborts for other cases.
531// Norm is C(l,eta) if one uses normalized wave functions, 1.0 if not.
532// So, one multiplies by C(l,eta) at the end if one uses normalized functions.
533// If there is overflow or underflow for C(l,eta) in this last case, one uses logs of F,F' and C(l,eta) for the calculation.
534//
535// Variables:
536// ----------
537// z : variable of the Coulomb wave function.
538// F,dF : regular wave function and derivative
539// z_square,z_two_eta,z_pow_l_plus_one : z^2, 2.z.eta, z^{l+1}.
540// n : index of the power series term. It begins at two.
541// an,an_minus_one,an_minus_two : a[n], a[n-1], a[n-2].
542// The test of convergence is |(n+l-1).a[n-2]|oo + |(n+l).a[n-1]|oo, as one of the two can be zero even before convergence.
543
544void Coulomb_wave_functions::F_dF_power_series (const std::complex<double> &z,std::complex<double> &F,std::complex<double> &dF)
545{
546 if (z == 0.0)
547 {
548 if (l == 0) F = 0.0,dF = (is_it_normalized) ? (Cl_eta) : (1.0);
549 else if (real (l) > 0) F = dF = 0.0;
550 else std::cout<<"F(z=0) and/or F'(z=0) are undefined."<<std::endl, abort ();
551 }
552 else
553 {
554 const std::complex<double> z_square = z*z,z_two_eta = 2.0*eta*z;
555
556 int n = 2;
557 std::complex<double> an_minus_two = 1.0,an_minus_one = z*eta/(l+1.0);
558
559 F = an_minus_two + an_minus_one;
560 dF = (l+1.0)*an_minus_two + (l+2.0)*an_minus_one;
561
562 while (inf_norm (an_minus_two*(n+l-1.0)) + inf_norm (an_minus_one*(n+l)) > precision)
563 {
564 const std::complex<double> an = (z_two_eta*an_minus_one - an_minus_two*z_square)/(n*(n + l + l + 1.0));
565
566 F += an;
567 dF += an*(n + l + 1.0);
568
569 n++;
570 an_minus_two = an_minus_one;
571 an_minus_one = an;
572 }
573
574 const std::complex<double> z_pow_l_plus_one = pow (z,l+1.0);

576 dF *= z_pow_l_plus_one/z;
577
578 if (is_it_normalized)
579 {
580 if ((Cl_eta == 0.0) || (!isfinite (Cl_eta))) F = exp (log_Cl_eta + log (F)),dF = exp (log_Cl_eta + log (dF));

582 }
583 }

592// Calculation of f = F'/F with a continued fraction.
593// --------------------------------------------------
594// One calculates the ratio f = F'/F with the continued fraction of the associated hypergeometric confluent function.
595// One uses Lentz's method.
596// One has : f = [b[0] + a[1]/b[1]+ a[2]/b[2]+ ... a[n]/b[n]+ ...]/z with :
597// b[0] = l + 1 + i.omega.z, a[n] = -2.i.omega.[1 + l + i.omega.eta] + (n-1).[-2.i.omega.z], b[n] = 2l + 2 + 2.i.omega.z + n-1.
598// omega is 1 or -1, and theoretically the result is the same.
599// If they are not equal numerically, omega = sign[-Im [z]] gives usually the best result.
600// If 1+l+i.omega.eta is a negative integer, f[omega] is finite and must be used.
601//
602// Variables:
603// ----------
604// z : variable of the Coulomb wave function.
605// omega : 1 or -1. Both values should be tried to test stability.
606// large,small : 1E50,1E-50. They are used in the case of vanishing denominators or numerators.
607// I_omega,a,b : i.omega, 1 + l + i.omega.eta, 2l + 2.
608// minus_two_I_omega_z,minus_two_I_omega_a_z,b_plus_two_I_omega_z : -2.i.omega.z, -2.i.omega.a.z, b + 2.i.omega.z
609// b0,Cn,Dn,an,bn,Delta_n : variables used in the Lentz method.
610// n,nm1 : index of a[n] and b[n], n-1
611// bn_plus_an_Dn : bn + an.Dn. Dn = 1/[bn + an.Dn] or large if bn + an.Dn is zero.
612// bn_plus_an_over_Cn : bn + an/Cn. Cn = bn + an/Dn or small if bn + an/Dn is zero.
613// fn : value of the continuous fraction during the iteration process.
614// test : test of convergence of fn.
615// f : value of F'(z)/F(z).
616
617std::complex<double> Coulomb_wave_functions::continued_fraction_f (const std::complex<double> &z,const int omega)
618{
619 const double small = 1E-50,large = 1E50;
620
621 const std::complex<double> I_omega(0.0,omega);
622 const std::complex<double> a = I_omega*eta + l + 1.0,b = 2*l + 2;
623 const std::complex<double> minus_two_I_omega_z = -2.0*I_omega*z,minus_two_I_omega_a_z = minus_two_I_omega_z*a,b_plus_two_I_omega_z = b - minus_two_I_omega_z;
624
625 const std::complex<double> b0 = l + 1.0 + I_omega*z;
626 std::complex<double> fn = (b0 != 0.0) ? (b0) : (small), Cn = fn, Dn = 0.0;
627
628 int n = 1;
629 double test;
630 do
631 {
632 const int nm1 = n-1;
633 const std::complex<double> an = minus_two_I_omega_a_z + nm1*minus_two_I_omega_z;
634 const std::complex<double> bn = b_plus_two_I_omega_z + nm1;
635
636 const std::complex<double> bn_plus_an_Dn = bn + an*Dn,bn_plus_an_over_Cn = bn + an/Cn;
637
638 Dn = (bn_plus_an_Dn != 0.0) ? (1.0/bn_plus_an_Dn) : (large);
639 Cn = (bn_plus_an_over_Cn != 0.0) ? (bn_plus_an_over_Cn) : (small);
640
641 const std::complex<double> Delta_n = Dn*Cn;
642 fn *= Delta_n;
643 test = inf_norm (1.0 - Delta_n);
644 n++;
645 }
646 while (test > 1E-15);
647
648 const std::complex<double> f = fn/z;
649
650 return f;
651}

670// H-'(z) = exp[-i.[z - eta.log[2z] - l.Pi/2 + sigma(l,eta)]].[S-'(z) - S-(z).i.(1 - eta/z)] .

724// Calculation of F(z) and F'(z) by direct integration.
725// ----------------------------------------------------
726// To calculate F(z) and F'(z), one integrates numerically F''(z) = [l(l+1)/z^2 + 2.eta/z - 1].F(z)
727// starting from debut, F_debut = F(debut) and dF_debut = F'(debut).
728// One always has Re[z] >= 0.0, so there is no branch cut problem.
729// If z = debut, the previous values are returned.
730// The starting point come from the stored values debut, F_debut and dF_debut.
731// If debut = 0, one puts debut = z/|2z| and calculates F(debut) and F'(debut) with power series.
732// The step of the integration is (z - debut)/N_num, with N_num = [|z - debut|/min (0.1,10.turning_point)] + 1.
733// The value of min (0.1,10/turning_point) gives a smaller step when turning_point increases,
734// as calculations become there more difficult as |F| typically varies faster in this case.
735// The intermediates points are called z_aft. They go from debut to z, and (debut,F_debut,dF_debut) is put to {z_aft,F(z_aft),F'(z_aft)} at each step.
736// If |F| increases along the path, the integration is stable.
737// If it decreases, and if z_aft decreases in modulus or one does not integrate with constant argument (i.e. theta constant in z = |z|.exp[i.theta])
738// for Re[l] > -1, one reintegrates F(z) from debut = z/|2z|, as integration is usually stable at constant argument for Re[l] > -1.
739// Increase or decrease is known using the Taylor expansion of F near debut in z up to second order.
740// If one integrates with constant argument and |F| decreases,
741// one integrates backwards from z_aft to debut with the knowledge of f(z_aft) = F'(z_aft)/F(z_aft).
742// F'(z_aft)/F(z_aft) is given by the continued fraction formula.
743// One then obtains by direct integration C.F(debut) and C.F'(debut).
744// Knowing F_debut, one deduces F(z) = 1/C and F'(z) = f(z_aft).F(z).
745// If 1+l+i.omega.eta is a negative integer, f[omega] is finite and is used.
746// Otherwise, f(z_aft) is calculated with omega = 1 and -1. If they are equal up to precision, f(omega) is correct and used.
747// omega is chosen so Re[-2.i.omega.z] < 0,
748// for which the anomalous convergence phenomenon of Gautschi of f is the smallest (W. Gautschi, Math. Comp. Vol. 31 p.994).
749// If not, but |norm.F| < 0.1, one still has to use f[omega], as it is probably correct as F is the minimal solution, and also one has no other way to calculate F.
750// If |norm.F| > 0.1 in this case, one stops the procedure and F will be calculated from H+ and H-, given by direct integration and continued fraction formulae.
751// In this case, is_it_successful is put to false, and otherwise the integration worked and it is put to true.
752// Norm is 1.0 if one uses normalized wave functions, C(l,eta) if not.
753//
754// If F(z_aft) is not finite, one stops the integration and is_it_successful is put to false.
755//
756// Variables
757// ---------
758// z : variable of the Coulomb wave function.
759// F,dF : regular wave function and derivative to calculate.
760// is_it_successful : false is the calculation is unstable,
761// i.e. |F| > 0.1 decreasing on the integration path and f(omega) is not equal to f(-omega) up to precision, true if not.
762// ODE : reference to pointer ODE_ptr, which performs direct integration from one point to another.
763// step_abs : length of the integration step from z to debut. It is min (0.1,10/turning_point)
764// N_num : number of integrations to do from debut to z. It is |z - debut|/step_abs + 1.
765// step_num : std::complex step from debut to z. It is (z - debut)/N_num .
766// ll_plus_one,two_eta : l(l+1), 2.eta. They are used in the Taylor expansion test.
767// z_aft : next point in which F is calculated by direct integration . It is z - i*step_num, with i from N_num-1 to 0.
768// one_over_debut,log_F_debut_der,d2F_debut_over_F_debut: 1/debut, F'(debut)/F(debut), F''(debut)/F(debut). They are used in the Taylor expansion test.
769// ratio : debut/z. It is used to know if one integrates with constant argument or if |z_aft| increases.
770// If not and |F| increases and Re[l] > -1, one reintegrates from z/|2z|.
771// f_omega,f_minus_omega : continued fractions F'/F with omega = sign(-Im[z_aft]) and -omega.
772// fp,fm : continued fractions F'/F with omega = 1 or -1 used in the case of 1+l +/- i.eta negative integer, as continued fractions are finite in this case.
773// norm_functions : C(l,eta)^2 for unnormalized functions, 1 for normalized functions.
774// F_debut_not_normed,dF_debut_not_normed : unnormed regular wave function and derivative in debut,
775// after integration from the starting point {z_aft, 1.0, F'(z_aft)/F(z_aft)}.
776// Then, F(z_aft) = F_debut/F_debut_not_normed and F'(z_aft) = f[omega].F .
777
778void Coulomb_wave_functions::F_dF_direct_integration (const std::complex<double> &z,std::complex<double> &F,std::complex<double> &dF,bool &is_it_successful)
779{
780 if (z == debut) {F = F_debut,dF = dF_debut,is_it_successful = true; return;}
781 if (debut == 0.0) debut = 0.5*z/abs (z),F_dF_power_series (debut,F_debut,dF_debut);
782
783 const class ODE_integration &ODE = *ODE_ptr;
784 const double step_abs = std::min(0.1,10.0/turning_point);
785 const unsigned int N_num = static_cast<unsigned int> (rint (abs (z-debut)/step_abs) + 1);
786 const std::complex<double> step_num = (z - debut)/static_cast<double> (N_num),ll_plus_one = l*(l+1.0),two_eta = 2.0*eta;
787 const std::complex<double> norm_functions = (!is_it_normalized) ? (Cl_eta*Cl_eta) : (1.0);
788
789 for (unsigned int i = N_num-1 ; i <= N_num ; i--)
790 {
791 const std::complex<double> z_aft = z - i*step_num,one_over_debut = 1.0/debut,log_F_debut_der = dF_debut/F_debut;
792 const std::complex<double> d2F_debut_over_F_debut = (ll_plus_one*one_over_debut + two_eta)*one_over_debut - 1.0;
793
794 if (abs (1.0 + step_num*(log_F_debut_der + 0.5*step_num*d2F_debut_over_F_debut)) < 1.0)
795 {
796 const std::complex<double> ratio = debut/z;
797 if ((real (l) > -1.0) && ((std::abs (imag (ratio)) > precision) || (real (ratio) > 1.0)))
798 {debut = 0.0, F_dF_direct_integration (z,F,dF,is_it_successful); return;}
799
800 std::complex<double> F_debut_not_normed,dF_debut_not_normed;
801
802 if (neg_int_omega_one)
803 {
804 const std::complex<double> fp = continued_fraction_f (z_aft,1);
805 ODE (z_aft,1.0,fp,debut,F_debut_not_normed,dF_debut_not_normed);
806 F = F_debut/F_debut_not_normed;
807 dF = fp*F;
808 }
809 else if (neg_int_omega_minus_one)
810 {
811 const std::complex<double> fm = continued_fraction_f (z_aft,-1);
812 ODE (z_aft,1.0,fm,debut,F_debut_not_normed,dF_debut_not_normed);
813 F = F_debut/F_debut_not_normed;
814 dF = fm*F;
815 }
816 else
817 {
818 const std::complex<double> f_omega = continued_fraction_f (z_aft,SIGN(-imag(z_aft))),f_minus_omega = continued_fraction_f (z_aft,-SIGN(-imag(z_aft)));
819
821 if ((abs (F*norm_functions) > 0.1) && (abs (f_minus_omega/f_omega - 1.0) > precision)) {is_it_successful = false; return;}
822
823 ODE (z_aft,1.0,f_omega,debut,F_debut_not_normed,dF_debut_not_normed);
824 F = F_debut/F_debut_not_normed;
825 dF = f_omega*F;
826 }
827 }
828 else ODE (debut,F_debut,dF_debut,z_aft,F,dF);
829
830 debut = z_aft,F_debut = F,dF_debut = dF;
831
832 if (!isfinite (F) || !isfinite (dF)) std::cout<<"Numerical failure encountered in F_dF_direct_integration."<<std::endl,exit (1);
833 }
834 is_it_successful = true;
835}

842// F(l,eta,z) = -F(l,-eta,-z).exp[-Pi.(eta-i.l)], F'(l,eta,z) = F'(l,-eta,-z).exp[-Pi.(eta-i.l)] if arg (z) > 0 and is_it_normalized is true,
843// F(l,eta,z) = -F(l,-eta,-z).exp[-Pi.(eta+i.l)], F'(l,eta,z) = F'(l,-eta,-z).exp[-Pi.(eta+i.l)] if arg (z) <= 0 and is_it_normalized is true,
844// F(l,eta,z) = -F(l,-eta,-z).exp[i.Pi.l)], F'(l,eta,z) = F'(l,-eta,-z).exp[i.Pi.l)] if arg (z) > 0 and is_it_normalized is false,
845// F(l,eta,z) = -F(l,-eta,-z).exp[-i.Pi.l)], F'(l,eta,z) = F'(l,-eta,-z).exp[-i.Pi.l)] if arg (z) <= 0 and is_it_normalized is false.
846//
847// F(l,-eta,-z) is calculated using the class cwf_minus_eta_ptr defined with (l,-eta).
848// The debut point of the class cwf_minus_eta_ptr is initialized with {debut,F_debut,dF_debut} and previous relations
849// if cwf_minus_eta_ptr->debut and -debut are different and debut non zero.
850//
851// If the normalization constant underflows or overflows, one uses logs.
852//
853// Variables
854// ---------
855// z : variable of the Coulomb wave function.
856// F,dF : regular wave function and derivative to calculate.
857// arg_debut, arg_z : argument angles of debut and z.
858// sym_constant_debut,log_sym_constant_debut : constant C so F(l,-eta,-debut) = C.F(l,eta,debut), its log.
859// sym_constant,log_sym_constant : constant C so F(l,eta,z) = C.F(l,-eta,-z), its log.

861void Coulomb_wave_functions::F_dF_with_symmetry_relations (const std::complex<double> &z,std::complex<double> &F,std::complex<double> &dF)
862{

864
865 if ((debut != 0.0) && (cwf_minus_eta_ptr->debut != -debut))
866 {
867 const double arg_debut = arg (debut);
868 const std::complex<double> sym_constant_debut = (arg_debut <= 0.0) ? (1.0/sym_constant_arg_neg) : (1.0/sym_constant_arg_pos);
869
870 cwf_minus_eta_ptr->debut = -debut;
871
872 if ((sym_constant_debut == 0.0) || (!isfinite (sym_constant_debut)))
873 {
874 const std::complex<double> log_sym_constant_debut = (arg (debut) <= 0.0) ? (-log_sym_constant_arg_neg) : (-log_sym_constant_arg_pos);
875
876 cwf_minus_eta_ptr->F_debut = exp (log_sym_constant_debut + log (F_debut));
877 cwf_minus_eta_ptr->dF_debut = -exp (log_sym_constant_debut + log (dF_debut));
878 }
879 else
880 {
881 cwf_minus_eta_ptr->F_debut = F_debut*sym_constant_debut;
882 cwf_minus_eta_ptr->dF_debut = -dF_debut*sym_constant_debut;
883 }
884 }

888
889 cwf_minus_eta_ptr->F_dF (-z,F,dF);

891 if ((sym_constant == 0.0) || (!isfinite (sym_constant)))
892 {
893 const std::complex<double> log_sym_constant = (arg_z <= 0.0) ? (log_sym_constant_arg_neg) : (log_sym_constant_arg_pos);
894
895 F = exp (log_sym_constant + log (F));
896 dF = -exp (log_sym_constant + log (dF));
897 }
898 else
899 {

909// Calculation of the asymptotic series
910// ------------------------------------
911// Asymptotic expansion:
912// S(+/-)(z) = 1.0+\sum_(n=1)^N a[n] with a[n+1]=a[n].[n.[n+1+/-2i.eta]+i.eta.(i.eta+/-1)-l(l+1)]/[+/-2i.(n+1)]/z, n >= 0 and a[0] = 1.
913// This expansion diverges : it is only useful with the smallest term summation method.
914// The test of convergence is max(|a[n]|oo,|n.a[n]/z|oo), so the largest norm of the term of series of function and derivative.
915// Practically, one stops when test < precision (it worked) or when test is not finite (it failed).
916// After that, one tests the series with the wronskian of H[omega] and H[-omega].
917//
918// Variables :
919// -----------
920// z is the variable of the Coulomb wave function.
921// omega : 1 if one calculates H+(z) and H+'(z), -1 if one calculates H-(z) and H-'(z).
922// one_over_z : 1/z.
923// sum : sum[0] is the series in H(omega), sum[1] the one in H(-omega).
924// dsum : dsum[0] is the series in H'(omega), dsum[1] the one in H'(-omega).
925// is_it_successful : true if the series converged and |wronskian - 2.i.omega|oo is smaller than precision, false if not.
926// test : test of convergence of the asymptotic series. It is max(|a[n]|oo,|n.a[n]/z|oo).
927// sign : if i=0, it is omega, if i=1, it is -omega.
928// Ieta,two_I_eta_sign : i.eta, 2.i.eta if sign = 1, -2.i.eta if sign = -1.
929// Ieta_Ieta_plus_sign_minus_ll_plus_one : i.eta(i.eta+1) - l(l+1) if sign=1, i.eta(i.eta-1) - l(l+1) if sign=-1.
930// n,an_sign : index of the series, a[n+1]
931// n_plus_one,sign_one_over_two_I_n_plus_one : n+1, 1/(2.i.(n+1)) if sign=1, -1/(2.i.(n+1)) if sign=-1.
932// sum_term, dsum_term : an_sign, (n+1).an_sign/z
933// two_I_omega : 2.i.omega .
934
935void Coulomb_wave_functions::asymptotic_series (const int omega,const std::complex<double> &one_over_z,
936 std::complex<double> sum[],std::complex<double> dsum[],bool &is_it_successful)
937{
938 sum[0] = sum[1] = 1.0;
939 dsum[0] = dsum[1] = 0.0;
940
941 double test;
942
943 for (unsigned int i = 0 ; i <= 1 ; i++)
944 {
945 const int sign = (i == 0) ? (omega) : (-omega);
946 const std::complex<double> Ieta(-imag (eta),real (eta)),two_I_eta_sign = 2.0*sign*Ieta,Ieta_Ieta_plus_sign_minus_ll_plus_one = Ieta*(Ieta + sign) - l*(l+1.0);
947
948 int n = 0;
949 std::complex<double> an_sign = 1.0;
950
951 do
952 {
953 const double n_plus_one = n + 1.0;
954 const std::complex<double> sign_one_over_two_I_n_plus_one(0,-sign*0.5/n_plus_one);
955
956 an_sign *= one_over_z*(n*(n_plus_one + two_I_eta_sign) + Ieta_Ieta_plus_sign_minus_ll_plus_one)*sign_one_over_two_I_n_plus_one;
957
958 const std::complex<double> sum_term = an_sign,dsum_term = n_plus_one*an_sign*one_over_z;
959
960 sum[i] += sum_term;
961 dsum[i] -= dsum_term;
962
963 test = std::max (inf_norm (sum_term),inf_norm (dsum_term));
964 n++;
965 }
966 while ((test > precision) && (isfinite (test)));
967
968 if (!isfinite (test)) {is_it_successful = false; return;}
969 }
970
971 const std::complex<double> two_I_omega(0.0,2.0*omega);
972 is_it_successful = (inf_norm (sum[1]*dsum[0] - sum[0]*dsum[1] + two_I_omega*(1.0 - eta*one_over_z)*sum[0]*sum[1] - two_I_omega) < precision);
973}

983// Numerical partial derivatives according to l or eta.
984// ----------------------------------------------------
985// One calculates here the partial derivatives according to l or eta
986// of the Coulomb wave functions F(x) or G(x) and of their derivatives with x F'(x) or G'(x),
987// with l_r=Re[l] and eta_r=Re[eta].
988// One considers here the parameters l_r and eta_r with the argument x.
989// For this, one uses the standard formula : df/d_chi(x) = [f(x,chi+eps) - f(x,chi-eps)]/[2.eps],
990// with chi = l_r or eta_r and eps = prec_first_order_expansion*chi. f is either F, F', G or G'.
991// One then calculates F or G, F' or G' with chi+eps and chi-eps.
992// With them, one obtains dF/d_chi(x),dF'/d_chi(x) or dG/d_chi(x),dG'/d_chi(x), with chi = l or eta.
993// All functions are real, so double values are returned, taking the real part of std::complex variables.
994//
995// Variables
996// ---------
997// is_it_regular : true if one calculates derivatives of F(z),F'(z), false if one calculates derivatives of G(z),G'(z).
998// is_it_eta : true if one calculates the partial derivatives according to eta,
999// false if one calculates the partial derivatives according to l.
1000// cwf_plus,cwf_minus : references on class Coulomb_wave_functions with parameters l+eps,eta and l-eps,eta or l,eta+eps and l,eta-eps.
1001// If one calculates the partial derivatives according to eta, they are *cwf_real_eta_plus_ptr and *cwf_real_eta_minus_ptr.
1002// If one calculates the partial derivatives according to l, they are *cwf_real_l_plus_ptr and *cwf_real_l_minus_ptr.
1003// x : Re[z], z the argument of the wave function.
1004// d_chi_Bx,d_chi_dBx : partial derivatives of F and F' in l_r,eta_r,x according to chi = l or eta if is_it_regular is true, or G and G' if not.
1005// l_r,eta_r,chi_r : Re[l],Re[eta], Re[eta] is is_it_eta is true, Re[l] if not.
1006// chi_r_plus,chi_r_minus : chi_r+eps, chi_r-eps.
1007// A_plus,dA_plus,A_minus,dA_minus : F(x,chi+eps),F'(x,chi+eps),F(x,chi-eps),F'(x,chi-eps) if is_it_regular is true,
1008// H+(x,chi+eps),H+'(x,chi+eps),H+(x,chi-eps),H+'(x,chi-eps) if is_it_regular is false.
1009// B_plus,B_minus,dB_plus,dB_minus : F(x,chi+eps),F'(x,chi+eps),F(x,chi-eps),F'(x,chi-eps) if is_it_regular is true,
1010// G(x,chi+eps),G'(x,chi+eps),G(x,chi-eps),G'(x,chi-eps) if is_it_regular is false.
1011

1015 const double chi_r_plus = (chi_r != 0.0) ? (chi_r*(1.0 + prec_first_order_expansion)) : (prec_first_order_expansion);
1016 const double chi_r_minus = (chi_r != 0.0) ? (chi_r*(1.0 - prec_first_order_expansion)) : (-prec_first_order_expansion);
1017
1018 if (is_it_eta)
1019 {
1020 if (cwf_real_eta_plus_ptr == 0) cwf_real_eta_plus_ptr = new class Coulomb_wave_functions (is_it_normalized,l_r,chi_r_plus);
1021 if (cwf_real_eta_minus_ptr == 0) cwf_real_eta_minus_ptr = new class Coulomb_wave_functions (is_it_normalized,l_r,chi_r_minus);
1022 }
1023 else
1024 {
1025 if (cwf_real_l_plus_ptr == 0) cwf_real_l_plus_ptr = new class Coulomb_wave_functions (is_it_normalized,chi_r_plus,eta_r);
1026 if (cwf_real_l_minus_ptr == 0) cwf_real_l_minus_ptr = new class Coulomb_wave_functions (is_it_normalized,chi_r_minus,eta_r);

1030 class Coulomb_wave_functions &cwf_minus = (is_it_eta) ? (*cwf_real_eta_minus_ptr) : (*cwf_real_l_minus_ptr);

1058// Calculation of F(z),F'(z) or G(z),G'(z) with the first order expansion method.
1059// ------------------------------------------------------------------------------
1060// When imaginary parts of l,eta,z are much smaller than their real parts but not all zero, with Re[z] > 0,
1061// one has to separate the calculation of the real and imaginary parts of (F,G)(z) and (F',G')(z), as they can differ by tens of orders of magnitude.
1062// Re[z] < 0 is not considered, as G(z) is std::complex with Im[z] = Im[l] = Im[eta] = 0.
1063// So, with z = x+i.y, eta = eta_r + i.eta_i and l = l_r+i.l_i, one calculates (F,G)[l_r,eta_r](x) and (F,G)'[l_r,eta_r](x).
1064//
1065// One considers here the parameters l_r and eta_r with the argument x, and the parameters l and eta with the argument z.
1066//
1067// One then has F(x),F'(x) from usual relations and one takes their real parts.
1068// One also has H+(x),H+'(x) from usual relations and G(x) = Re[H+(x)],G'(x) = Re[H+'(x)].
1069//
1070// After that, one expands (F,G)(z) and (F',G')(z) in first order in y, eta_i and l_i and one has, up to y^2,eta_i^2 and l_i^2:
1071//
1072// (F,G)(z) = (F,G)(x) + i.y.d(F,G)/dx[omega](x) + i.eta_i.d(F,G)/d[eta](x) + i.l_i.d(F,G)/dl(x).
1073// (F',G')[omega](z) = (F',G')[omega](x) + i.y.(F'',G'')[omega](x) + i.eta_i.d(F',G')/d[eta](x) + i.l_i.d(F',G')/dl(x).
1074//
1075// Variables:
1076// ----------
1077// is_it_regular : true if one calculates F(z),F'(z), false if one calculates G(z),G'(z).
1078// z : variable of the Coulomb wave function.
1079// B,dB : F(z),F'(z) if is_it_regular is true, G(z),G'(z) if it is false.
1080// x,y,l_r,l_i,eta_r,eta_i : Re[z], Im[z], Re[l], Im[l], Re[eta], Im[eta].
1081// A_x,dA_x : F(x),F'(x) if is_it_regular is true, H+(x),H+'(x) if it is false.
1082// Bx,dBx : F(x),F'(x) if is_it_regular is true, G(x),G'(x) if it is false.
1083// cwf_real : class Coulomb_wave_functions of parameters l_r and eta_r.
1084// one_over_x,two_eta_r : 1/x, 2.eta_r.
1085// d_l_Bx,d_l_dBx,d_eta_Bx,d_eta_dBx : partial derivatives according to l and eta of B(x) and B'(x)
1086// They are initialized at zero, and not calculated if they are multiplied by zero after.
1087// Coulomb_eq_x,d2Bx : l_r(l_r+1)/[x^2] + 2.eta_r/x - 1, B''(x) = [l_r(l_r+1)/[x^2] + 2.eta_r/x - 1].B(x),
1088
1089
1090void Coulomb_wave_functions::first_order_expansions (const bool is_it_regular,const std::complex<double> &z,std::complex<double> &B,std::complex<double> &dB)
1091{
1092 const double x = real (z),y = imag (z),l_r = real (l),l_i = imag (l),eta_r = real (eta),eta_i = imag (eta);
1093
1094 if (cwf_real_ptr == 0) cwf_real_ptr = new class Coulomb_wave_functions (is_it_normalized,l_r,eta_r);
1095 class Coulomb_wave_functions &cwf_real = *cwf_real_ptr;
1096
1097 std::complex<double> A_x,dA_x;
1098 if (is_it_regular)
1099 cwf_real.F_dF (x,A_x,dA_x);
1100 else
1101 cwf_real.H_dH (1,x,A_x,dA_x);
1102
1103 const double Bx = real (A_x),dBx = real (dA_x);
1104
1105 double d_l_Bx = 0.0,d_l_dBx = 0.0,d_eta_Bx = 0.0,d_eta_dBx = 0.0;
