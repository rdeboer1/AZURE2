89bool parseOptions(int argc, char *argv[], Config& configure) {
90#ifndef NO_READLINE
91 bool useReadline=true;
92#else
93 bool useReadline = false;
94#endif
95 configure.configfile="";
96 std::vector<std::string> options;
97 for(int i=1;i<argc;i++) {
98 std::string arg=argv[i];
99 if(!arg.empty()&&arg.substr(0,2)=="--") options.push_back(arg);
100 else configure.configfile=arg;
101 }
102 char* optionsFile = getenv("AZURE_OPTIONS_FILE");
103 if(optionsFile) {
104 std::ifstream in(optionsFile);
105 if(in) {
106 while(!in.eof()) {
107 std::string newOption;
108 getline(in,newOption);
109 if(!in.eof()) {
110 std::string trimmedOption="";
111 for(std::string::iterator it = newOption.begin();it<newOption.end();it++)
112 if(*it!=' '&&*it!='\t'&&*it!='\n')
113 trimmedOption+=*it;
114 if(!trimmedOption.empty()&&trimmedOption.substr(0,2)=="--") options.push_back(trimmedOption);
115 }
116 }
117 in.close();
118 } else configure.outStream << "AZURE_OPTIONS_FILE variable set, but file not readable." << std::endl;
119 }
120 for(std::vector<std::string>::iterator it = options.begin();it<options.end();it++) {
121#ifndef NO_READLINE
122 if(*it=="--no-readline") useReadline=false;
123 else if(*it=="--no-transform") configure.paramMask &= ~Config::TRANSFORM_PARAMETERS;
124#else
125 if (*it == "--no-transform") configure.paramMask &= ~Config::TRANSFORM_PARAMETERS;
126#endif
127 else if(*it=="--no-long-wavelength") configure.paramMask &= ~Config::USE_LONGWAVELENGTH_APPROX;
128 else if(*it=="--use-brune") configure.paramMask |= Config::USE_BRUNE_FORMALISM;
129 else if(*it=="--gsl-coul") configure.paramMask |= Config::USE_GSL_COULOMB_FUNC;
130 else if(*it=="--ignore-externals") configure.paramMask |= Config::IGNORE_ZERO_WIDTHS;
131 else if(*it=="--use-rmc") configure.paramMask |= Config::USE_RMC_FORMALISM;
132 else if(*it=="--no-gui") continue;
133 else configure.outStream << "WARNING: Unknown option " << *it << '.' << std::endl;
134 }
135 return useReadline;
136}

247bool readSegmentFile(const Config& configure,std::vector<SegPairs>& segPairs) {
248 std::ifstream in;
249 std::string startTag,stopTag;
250 if(configure.paramMask & Config::CALCULATE_WITH_DATA) {
251 startTag="<segmentsData>";
252 stopTag="</segmentsData>";
253 } else {
254 startTag="<segmentsTest>";
255 stopTag="</segmentsTest>";
256 }
257 in.open(configure.configfile.c_str());
258 if(in) {
259 std::string line="";
260 while(line!=startTag&&!in.eof()) getline(in,line);
261 if(line==startTag) {
262 line="";
263 while(line!=stopTag&&!in.eof()) {
264 getline(in,line);
265 bool empty=true;
266 for(unsigned int i=0;i<line.size();++i)
267 if(line[i]!=' '&&line[i]!='\t') {
268 empty=false;
269 break;
270 }
271 if(empty==true) continue;
272 if(line!=stopTag&&!in.eof()) {
273 std::istringstream stm;
274 stm.str(line);
275 if(configure.paramMask & Config::CALCULATE_WITH_DATA) {
276 SegLine segment(stm);
277 if(!(stm.rdstate() & (std::stringstream::failbit | std::stringstream::badbit))&&segment.isActive()==1) {
278 SegPairs tempSet={segment.entranceKey(),segment.exitKey()};
279 segPairs.push_back(tempSet);
280 }
281 } else {
282 ExtrapLine segment(stm);
283 if(!(stm.rdstate() & (std::stringstream::failbit | std::stringstream::badbit))&&segment.isActive()==1) {
284 SegPairs tempSet={segment.entranceKey(),segment.exitKey()};
285 segPairs.push_back(tempSet);
286 }
287 }
288 }
289 }
290 if(line!=stopTag) {
291 configure.outStream << "Problem reading segments. Check configuration file." << std::endl;
292 return false;
293 }
294 } else {
295 configure.outStream << "Problem reading segments. Check configuration file." << std::endl;
296 return false;
297 }
298 in.close();
299 } else {
300 configure.outStream << "Cannot read segments. Check configuration file." << std::endl;
301 return false;
302 }
303 in.clear();
304 return true;
305}

438bool checkExternalCapture(Config& configure, const std::vector<SegPairs>& segPairs) {
439 std::ifstream in;
440 in.open(configure.configfile.c_str());
441 if(!in) {
442 configure.outStream << "Cannot read nuclear information. Check configuration file." << std::endl;
443 return false;
444 }
445 std::string line="";
446 while(line!="<levels>"&&!in.eof()) getline(in,line);
447 if(line!="<levels>") {
448 configure.outStream << "Cannot read nuclear information. Check configuration file." << std::endl;
449 return false;
450 }
451 line="";
452 while(line!="</levels>"&&!in.eof()&&
453 !(configure.paramMask & Config::USE_EXTERNAL_CAPTURE)) {
454 getline(in,line);
455 bool empty=true;
456 for(unsigned int i=0;i<line.size();++i)
457 if(line[i]!=' '&&line[i]!='\t') {
458 empty=false;
459 break;
460 }
461 if(empty==true) continue;
462 if(line!="</levels>"&&!in.eof()) {
463 std::istringstream stm;
464 stm.str(line);
465 NucLine tempNucLine(stm);
466 if((stm.rdstate() & (std::stringstream::failbit | std::stringstream::badbit))) {
467 configure.outStream << "Problem reading nuclear information. Check configuration file." << std::endl;
468 return false;
469 }
470 if(tempNucLine.ecMultMask()!=0) {
471 for(int i=0;i<segPairs.size();i++) {
472 if(tempNucLine.ir()==segPairs[i].secondPair||
473 segPairs[i].secondPair==-1) {
474 configure.paramMask |= Config::USE_EXTERNAL_CAPTURE;
475 break;
476 }
477 }
478 }
479 }
480 }
481 in.close();
482 in.clear();
483 if((configure.paramMask & Config::USE_EXTERNAL_CAPTURE)&&
484 (configure.paramMask & Config::USE_RMC_FORMALISM)) {
485 configure.outStream << "WARNING: External capture is not compatible with Reich-Moore. Ignoring external capture."
486 << std::endl;
487 configure.paramMask &= ~Config::USE_EXTERNAL_CAPTURE;
488 }
489 return true;
490}

556int main(int argc,char *argv[]){
557
558 //Check for --help option first. If set, print help and exit.
559 for(int i=1;i<argc;i++)
560 if(strcmp(argv[i],"--help")==0) {
561 printHelp();
562 return 0;
563 }
564
565 //Set GSL Error Handler
566 gsl_set_error_handler (&GSLException::GSLErrorHandler);
567
568 //If GUI is built, look for --no-gui option. If not set, hand control to GUI.
569#ifdef GUI_BUILD
570 bool useGUI=true;
571 for(int i=1;i<argc;i++)
572 if(strcmp(argv[i],"--no-gui")==0) useGUI=false;
573 if(useGUI) return start_gui(argc,argv);
574#endif
575
576 //Create new configuration structure, and parse the command line parameters
577 Config configure(std::cout);
578 bool useReadline = parseOptions(argc,argv,configure);
579
580 //Read the parameters from the runtime configuration file
581 if(configure.configfile.empty()) {
582 configure.outStream << "A valid configuration file must be specified." << std::endl
583 << "\tSyntax: AZURE2 <options> configfile" << std::endl;
584 return -1;
585 }
586 if(configure.ReadConfigFile()==-1) {
587 configure.outStream << "Could not open " << configure.configfile << ". Check that file exists."
588 << std::endl;
589 return -1;
590 }
591#ifndef NO_STAT
592 else if(configure.CheckForInputFiles() == -1) return -1;
593#endif
594 if((configure.paramMask & Config::USE_RMC_FORMALISM) && (configure.paramMask & Config::USE_BRUNE_FORMALISM)) {
595 configure.outStream << "WARNING: --use-brune is incompatible with --use-rmc. Ignoring --use-brune." << std::endl;
596 configure.paramMask &= ~Config::USE_BRUNE_FORMALISM;
597 }
598 if((configure.paramMask & Config::USE_BRUNE_FORMALISM)||(configure.paramMask & Config::IGNORE_ZERO_WIDTHS)) {
599 if(!(configure.paramMask & Config::USE_AMATRIX))
600 configure.outStream << "WARNING: R-Matrix specified but --ignore-externals and --use-brune options require A-Matrix. A-Matrix will be used."
601 << std::endl;
602 configure.paramMask |= Config::USE_AMATRIX;
603 }
604
605 //Print welcome message
606 welcomeMessage(configure);
607
608 //Read and process command, setting appropriate configuration flags
609 processCommand(commandShell(configure),configure);
610
611 //Open history file for readline
612#ifndef NO_READLINE
613 if(useReadline) read_history("./.azure_history");
614#endif
615
616 //Read the external parameter file to be used, if any
617 getParameterFile(useReadline,configure);
618
619 //Parse the segment files for entrance,exit pairs
620 std::vector<SegPairs> segPairs;
621 if(!(configure.paramMask & Config::CALCULATE_REACTION_RATE)) {
622 if(!readSegmentFile(configure,segPairs)) exit(1);
623 } else getRateParams(configure,segPairs,useReadline);
624
625 //Check if the entrance,exit pairs are in the external capture file
626 // If so, external capture will be needed
627 if(!checkExternalCapture(configure,segPairs)) exit(1);
628
629 //Read the external capture file name to be used, if any
630 getExternalCaptureFile(useReadline,configure);
631
632 //Create instance of main AZURE function, print start message,
633 // and execute
634 AZUREMain azureMain(configure);
635 configure.outStream << std::endl; startMessage(configure);
636 int returnValue = azureMain();
637
638 //Print exit message
639 exitMessage(configure);
640
641 //Write readline history file
642#ifndef NO_READLINE
643 if(useReadline) write_history("./.azure_history");
644#endif
645
646 return returnValue;
647}
