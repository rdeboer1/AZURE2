507void EPoint::ConvertLabAngle(PPair *entrancePair, PPair *exitPair, const Config& configure) {
508 double qValue=entrancePair->GetSepE()+entrancePair->GetExE()-exitPair->GetSepE()-exitPair->GetExE();
509 double a13=(entrancePair->GetM(1)*exitPair->GetM(1))*this->GetLabEnergy()/(this->GetLabEnergy()+qValue)/
510 (entrancePair->GetM(1)+entrancePair->GetM(2))/(exitPair->GetM(1)+exitPair->GetM(2));
511 double a24=(entrancePair->GetM(2)*exitPair->GetM(2))*(1+entrancePair->GetM(1)/entrancePair->GetM(2)*qValue/(this->GetLabEnergy()+qValue))/
512 (entrancePair->GetM(1)+entrancePair->GetM(2))/(exitPair->GetM(1)+exitPair->GetM(2));
513
514 if(a13>a24) {
515 double thetaMax=asin(sqrt(a24/a13))*180./pi;
516 if(thetaMax<this->GetLabAngle()) configure.outStream << std::endl << "Lab Angle (" << this->GetLabAngle()
517 << " degrees) is not kinematically possible. Maximum angle is "
518 << thetaMax << " degrees." << std::endl;
519 assert(thetaMax>=this->GetLabAngle());
520 }
521
522 double E3PerEt=a13*pow(cos(this->GetLabAngle()*pi/180.)+sqrt(a24/a13-pow(sin(this->GetLabAngle()*pi/180.),2.0)),2.0);
523 double tempE3PerEt = a13*pow(cos((this->GetLabAngle()+0.001)*pi/180.)+sqrt(a24/a13-pow(sin((this->GetLabAngle()+0.001)*pi/180.),2.0)),2.0);
524 double slope = sqrt(tempE3PerEt/a24)*sin((this->GetLabAngle()+0.001)*pi/180.)-sqrt(E3PerEt/a24)*sin(this->GetLabAngle()*pi/180.);
525 bool switchDomain=false;
526 if(slope<0.) switchDomain=true;
527
528 cm_angle_=180./pi*asin(sqrt(E3PerEt/a24)*sin(this->GetLabAngle()*pi/180.));
529 if(switchDomain) cm_angle_ = 180.- cm_angle_;
530}

538void EPoint::ConvertCrossSection(PPair *entrancePair, PPair *exitPair) {
539 double conversionFactor;
540 if(this->GetLabAngle()==0.0 || this->GetLabAngle()==180.0) {
541 double m1=entrancePair->GetM(1);
542 double m2=entrancePair->GetM(2);
543 double m3=exitPair->GetM(1);
544 double m4=exitPair->GetM(2);
545 double e1=this->GetLabEnergy();
546 double qValue=entrancePair->GetSepE()+entrancePair->GetExE()-exitPair->GetSepE()-exitPair->GetExE();
547 double et=e1+qValue;
548 double a=m1*m4*e1/(m1+m2)/(m3+m4)/et;
549 double b=m1*m3*e1/(m1+m2)/(m3+m4)/et;
550 double c=m2*m3/(m1+m2)/(m3+m4) * (1+m1*qValue/m2/et);
551 double d=m2*m4/(m1+m2)/(m3+m4) * (1+m1*qValue/m2/et);
552 double e3et=b+d+2*pow(a*c,0.5)*cos(pi/180.*this->GetCMAngle());
553 conversionFactor=pow(a*c,0.5)*pow(d/b-pow(sin(this->GetLabAngle()*pi/180.),2.0),0.5)/e3et;
554 }
555 else {
556 conversionFactor=pow(sin(pi/180.*this->GetLabAngle())/sin(pi/180.*this->GetCMAngle()),2.0)*cos(pi/180.*(this->GetCMAngle()-this->GetLabAngle()));
557 }
558 cm_crosssection_=this->GetLabCrossSection()*conversionFactor;
559 cm_dcrosssection_=this->GetLabCrossSectionError()*conversionFactor;
560}

606void EPoint::CalcLegendreP(int maxL,TargetEffect* targetEffect) {
607 double x=cos(this->GetCMAngle()*pi/180.0);
608 if(maxL>=0) {
609 if(targetEffect && targetEffect->NumQCoefficients()>0)
610 this->AddLegendreP(targetEffect->GetQCoefficient(0));
611 else this->AddLegendreP(1.0);
612 double polyMinusTwo=1.0;
613 if(maxL>=1) {
614 if(targetEffect && targetEffect->NumQCoefficients()>1)
615 this->AddLegendreP(x*targetEffect->GetQCoefficient(1));
616 else this->AddLegendreP(x);
617 double polyMinusOne=x;
618 if(maxL>=2) {
619 for(int lOrder=2;lOrder<=maxL;lOrder++) {
620 double poly=(2.0*lOrder-1.0)/lOrder*x*polyMinusOne-
621 (lOrder-1.0)/lOrder*polyMinusTwo;
622 if(targetEffect && targetEffect->NumQCoefficients()>lOrder)
623 this->AddLegendreP(poly*targetEffect->GetQCoefficient(lOrder));
624 else this->AddLegendreP(poly);
625 polyMinusTwo=polyMinusOne;
626 polyMinusOne=poly;
627 }
628 }
629 }
630 }
631 for(int i=1;i<=this->NumSubPoints();i++) {
632 this->GetSubPoint(i)->CalcLegendreP(maxL, targetEffect);
633 }

890 theECMGroup->IsChannelCapture());

894 this->AddECAmplitude(k,ecm,ecAmplitude);
895 }
896 }
897 }
898 }
899 }
900 }

1009void EPoint::IntegrateTargetEffect() {
1010 double yield=0.0;
1011 TargetEffect *targetEffect=this->GetParentData()->GetTargetEffect(this->GetTargetEffectNum());
1012 double energyStep=this->GetSubPoint(1)->GetCMEnergy()-this->GetSubPoint(2)->GetCMEnergy();
1013 if(targetEffect->IsConvolution()&&targetEffect->IsTargetIntegration()) {
1014 int outerLowerLimit=round((this->GetSubPoint(1)->GetCMEnergy()-this->GetCMEnergy())/energyStep)+1;
1015 int outerUpperLimit=outerLowerLimit-1+round(this->GetTargetThickness()/energyStep);
1016 double outerIntFirst=0.0;
1017 double outerIntEvenSum=0.0;
1018 double outerIntOddSum=0.0;
1019 double outerIntegral=0.0;
1020 int outerCounter=0;
1021 for(int i=outerLowerLimit;i<=outerUpperLimit;i++) {
1022 int innerLowerLimit;
1023 if(i-round(targetEffect->convolutionRange*targetEffect->GetSigma()/energyStep)>1)
1024 innerLowerLimit=i-round(targetEffect->convolutionRange*targetEffect->GetSigma()/energyStep);
1025 else innerLowerLimit=1;
1026 int innerUpperLimit;
1027 if(i+round(targetEffect->convolutionRange*targetEffect->GetSigma()/energyStep)-1<targetEffect->NumSubPoints())
1028 innerUpperLimit=i+round(targetEffect->convolutionRange*targetEffect->GetSigma()/energyStep)-1;
1029 else innerUpperLimit=targetEffect->NumSubPoints();
1030 double innerIntFirst=0.0;
1031 double innerIntEvenSum=0.0;
1032 double innerIntOddSum=0.0;
1033 double innerIntegral=0.0;
1034 double centroid=this->GetSubPoint(i)->GetCMEnergy();
1035 int innerCounter=0;
1036 for(int ii=innerLowerLimit;ii<=innerUpperLimit;ii++) {
1037 double thisEnergy=this->GetSubPoint(ii)->GetCMEnergy();
1038 double innerIntegrand=this->GetSubPoint(ii)->GetFitCrossSection()/
1039 this->GetSubPoint(ii)->GetStoppingPower()/1e24*targetEffect->GetConvolutionFactor(thisEnergy,centroid);
1040 if(innerCounter==0) innerIntFirst=innerIntegrand;
1041 else if(innerCounter%2==0) {
1042 innerIntEvenSum+=innerIntegrand;
1043 if(innerCounter>=2) innerIntegral=energyStep/3.0*(innerIntFirst+4.0*innerIntOddSum+2.0*innerIntEvenSum-innerIntegrand);
1044 } else if(innerCounter%2!=0) {
1045 innerIntOddSum+=innerIntegrand;
1046 if(innerCounter>=2) innerIntegral=energyStep/3.0*(innerIntFirst+4.0*innerIntOddSum+2.0*innerIntEvenSum-3.0*innerIntegrand);
1047 }
1048 innerCounter++;
1049 }
1050 if(outerCounter==0) outerIntFirst=innerIntegral;
1051 else if(outerCounter%2==0) {
1052 outerIntEvenSum+=innerIntegral;
1053 if(outerCounter>=2) outerIntegral=energyStep/3.0*
1054 (outerIntFirst+4.0*outerIntOddSum+2.0*outerIntEvenSum-innerIntegral);
1055 } else if(outerCounter%2!=0) {
1056 outerIntOddSum+=innerIntegral;
1057 if(outerCounter>=2) outerIntegral=energyStep/3.0*
1058 (outerIntFirst+4.0*outerIntOddSum+2.0*outerIntEvenSum-3.0*innerIntegral);
1059 }
1060 outerCounter++;
1061 }
1062 yield=outerIntegral;
1063 } else if(targetEffect->IsConvolution()) {
1064 double intFirst=0.0;
1065 double intEvenSum=0.0;
1066 double intOddSum=0.0;
1067 double integral=0.0;
1068 double centroid=this->GetCMEnergy();
1069 for(int i=0;i<this->NumSubPoints();i++) {
1070 double thisEnergy=this->GetSubPoint(i+1)->GetCMEnergy();
1071 double integrand=this->GetSubPoint(i+1)->GetFitCrossSection()
1072 *targetEffect->GetConvolutionFactor(thisEnergy,centroid);
1073 if(i==0) intFirst=integrand;
1074 else if(i%2==0) {
1075 intEvenSum+=integrand;
1076 if(i>=2) integral=energyStep/3.0*(intFirst+4.0*intOddSum+2.0*intEvenSum-integrand);
1077 } else if(i%2!=0) {
1078 intOddSum+=integrand;
1079 if(i>=2) integral=energyStep/3.0*(intFirst+4.0*intOddSum+2.0*intEvenSum-3.0*integrand);
1080 }
1081 }
1082 yield=integral;
1083 } else if(targetEffect->IsTargetIntegration()) {
1084 double intFirst=0.0;
1085 double intEvenSum=0.0;
1086 double intOddSum=0.0;
1087 double integral=0.0;
1088 for(int i=0;i<this->NumSubPoints();i++) {
1089 double thisEnergy=this->GetSubPoint(i+1)->GetCMEnergy();
1090 double integrand=this->GetSubPoint(i+1)->GetFitCrossSection()/
1091 this->GetSubPoint(i+1)->GetStoppingPower()/1e24;
1092 if(i==0) intFirst=integrand;
1093 else if(i%2==0) {
1094 intEvenSum+=integrand;
1095 if(i>=2) integral=energyStep/3.0*(intFirst+4.0*intOddSum+2.0*intEvenSum-integrand);
1096 } else if(i%2!=0) {
1097 intOddSum+=integrand;
1098 if(i>=2) integral=energyStep/3.0*(intFirst+4.0*intOddSum+2.0*intEvenSum-3.0*integrand);
1099 }
1100 }
1101 yield=integral;
1102 }
1103 this->SetFitCrossSection(yield);
1104}
