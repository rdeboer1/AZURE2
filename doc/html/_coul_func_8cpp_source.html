162double CoulFunc::PEShift(int l,double radius,double energy) {
163 struct CoulWaves coul=this->operator()(l,radius,energy);
164 double rho=sqrt(2.*uconv)/hbarc*radius*sqrt(redmass()*energy);
165 if(pow(coul.F,2.0)==0.&&coul.F*coul.dF==0.) return rho*(coul.dG/coul.G);
