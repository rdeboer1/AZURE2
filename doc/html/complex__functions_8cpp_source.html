3// Precise evaluation of exp[z]-1 for z complex
4// --------------------------------------------
5// When |Re[z]| >= 1 or |Im[z]| >= 1, one uses directly the standard exp function as it is precise.
6// Otherwise, numerical cancellations can occur.
7// So, one uses the always stable formula exp[z]-1 = expm1(x) - 2.exp(x).sin^2(y/2) + i.exp(x).sin(y)
8// with x = Re[z] and y = Im[z]. expm1(x) gives a precise evaluation of exp(x)-1 for x double.
9
10std::complex<double> expm1 (const std::complex<double> &z)
11{
12 const double x = real (z),y = imag (z);
13
14 if ((std::abs (x) >= 1.0) || (std::abs (y) >= 1.0)) return (exp (z) - 1.0);
15
16 const double expm1_x = expm1 (x),exp_x = 1.0 + expm1_x,sin_y_over_two = sin (0.5*y),sin_y = sin (y);
17
18 return std::complex<double> (expm1_x - 2.0*exp_x*sin_y_over_two*sin_y_over_two,exp_x*sin_y);
19}

23// Precise evaluation of log[1+z] for z complex
24// --------------------------------------------
25// When |Re[z]| >= 1 or |Im[z]| >= 1, one uses directly the standard log function as it is precise.
26// Otherwise, numerical cancellations can occur.
27// So, one uses the always stable formula log[1+z] = log1p(x) + log1p([y/(1+x)]^2)/2 + i.atan2(y,1+x)
28// with x = Re[z] and y = Im[z]. log1p(x) gives a precise evaluation of log(1+x) for x double.
29// atan2(x,y) gives the arc tangent of y/x so it is in ]-Pi:Pi].
30
31std::complex<double> log1p (const std::complex<double> &z)
32{
33 const double x = real (z),y = imag (z);
34
35 const double xp1 = 1.0 + x,abs_x = std::abs (x), abs_y = std::abs (y);
36
37 if ((abs_x >= 1.0) || (abs_y >= 1.0)) return log (1.0 + z);
38
39 const double y_over_xp1 = y/xp1;
40
41 return std::complex<double> (log1p (x) + 0.5*log1p (y_over_xp1*y_over_xp1),atan2 (y,xp1));
42}

47// Logarithm of Gamma[z], z anywhere in the complex plane except in the Gamma function poles.
48// ------------------------------------------------------------------------------------------
49// If z is not finite or is a negative integer, the program returns an error message and stops.
50// The Lanczos method is used. Precision : < 2E-10 in theory, < 1E-12 in almost every case.
51// The method works for Re[z] > 0.
52// If Re[z] <= 0, one uses the formula Gamma[z].Gamma[1-z] = Pi/sin (Pi.z).
53// log[sin(Pi.z)] is calculated with the Kolbig method (K.S. Kolbig, Comp. Phys. Comm., Vol. 4, p.221 (1972)) :
54// If z = x+iy and y >= 0, log[sin(Pi.z)] = log[sin(Pi.eps)] - i.Pi.n, with z = n + eps so 0 <= Re[eps] < 1 and n integer.
55// If y > 110, log[sin(Pi.z)] = -i.Pi.z + log[0.5] + i.Pi/2 numerically so that no overflow can occur.
56// If z = x+iy and y < 0, log[Gamma(z)] = [log[Gamma(z*)]]*, so that one can use the previous formula with z*.
57//
58// Variables:
59// ----------
60// z,z_p_0p5,z_p_5p5 : argument of the Gamma function, z+0.5, z+5.5
61// sqrt_2Pi,log_Pi : sqrt(2.Pi), log(Pi).
62// sum : Rational function in the Lanczos method.
63// log_Gamma_z : log[Gamma(z)] value.
64// c : table containing the seven coefficients in the expansion used in the Lanczos method.
65// eps : z = n + eps so 0 <= Re[eps] < 1 and n integer.
66// log_const : log[0.5] + i.Pi/2
67
68std::complex<double> log_Gamma (const std::complex<double> &z)

73
74 if ((z == rint (x)) && (x <= 0)) std::cout<<"z is negative integer in log_Gamma."<<std::endl, abort ();

76 if (x > 0.0)
77 {
78 const std::complex<double> z_p_0p5 = z + 0.5, z_p_5p5 = z + 5.5;

80 const double c[7] = {1.000000000190015,
81 7.618009172947146E+1,
82 -8.650532032941677E+1,

112// Coulomb phase shift.
113// --------------------
114// It is given by the formula [Gamma[1+l+I.eta] - Gamma[1+l-I.eta]]/[2i].
115// 0 is returned if 1+l+/-I.eta is a negative integer.
116//
117// Variables:
118// ----------
119// l : orbital angular momentum l.
120// eta : Sommerfeld parameter.
121// Ieta,one_over_two_I : i.eta,1/[2.i] .
122// arg_plus,arg_minus : 1+l+i.eta, 1+l-i.eta.
123// log_Gamma_plus,log_Gamma_minus : logs of Gamma[1+l+I.eta], Gamma[1+l-I.eta].
124// sigma_l : returned result.
125
126std::complex<double> sigma_l_calc (const std::complex<double> &l,const std::complex<double> &eta)
127{
128 const std::complex<double> Ieta(-imag (eta),real (eta)),one_over_two_I(0,-0.5);
129 const std::complex<double> arg_plus = 1.0 + l + Ieta, arg_minus = 1.0 + l - Ieta;
130
131 if ((rint (real (arg_plus)) == arg_plus) && (rint (real (arg_plus)) <= 0.0)) return 0.0;
132 if ((rint (real (arg_minus)) == arg_minus) && (rint (real (arg_minus)) <= 0.0)) return 0.0;
133
134 const std::complex<double> log_Gamma_plus = log_Gamma (arg_plus),log_Gamma_minus = log_Gamma (arg_minus);
135 const std::complex<double> sigma_l = (log_Gamma_plus - log_Gamma_minus)*one_over_two_I;
136
137 return sigma_l;
138}
