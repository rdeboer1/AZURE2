783void AZURESetup::SaveAndRun() {
784 save();
785 if(GetConfig().configfile.empty()) return;
786 runTab->runtimeText->clear();
787 QFile file(QString::fromStdString(GetConfig().configfile));
788 QFileInfo info(file);
789 QString directory=info.absolutePath();
790 if(GetConfig().outputdir.empty()) GetConfig().outputdir=QDir::fromNativeSeparators(directory).toStdString()+'/';
791 if(GetConfig().checkdir.empty()) GetConfig().checkdir=QDir::fromNativeSeparators(directory).toStdString()+'/';
792
793 GetConfig().chiVariance=runTab->chiVarianceText->text().toDouble();
794
795 if(runTab->calcType->currentIndex()==1 ||
796 runTab->calcType->currentIndex()==3) GetConfig().paramMask |= Config::PERFORM_FIT;
797 else GetConfig().paramMask &= ~Config::PERFORM_FIT;
798 if(runTab->calcType->currentIndex()==2||
799 runTab->calcType->currentIndex()==4) GetConfig().paramMask &= ~Config::CALCULATE_WITH_DATA;
800 else GetConfig().paramMask |= Config::CALCULATE_WITH_DATA;
801 if(runTab->calcType->currentIndex()==3) GetConfig().paramMask |= Config::PERFORM_ERROR_ANALYSIS;
802 else GetConfig().paramMask &= ~Config::PERFORM_ERROR_ANALYSIS;
803 if(runTab->calcType->currentIndex()==4) GetConfig().paramMask |= Config::CALCULATE_REACTION_RATE;
804 else GetConfig().paramMask &= ~Config::CALCULATE_REACTION_RATE;
805
806 if(runTab->oldParamFileButton->isChecked()) {
807 GetConfig().paramMask |= Config::USE_PREVIOUS_PARAMETERS;
808 GetConfig().paramfile=runTab->paramFileText->text().toStdString();
809 } else GetConfig().paramMask &= ~Config::USE_PREVIOUS_PARAMETERS;
810
811 std::vector<SegPairs> segPairs;
812 if(!(GetConfig().paramMask & Config::CALCULATE_REACTION_RATE)) {
813 if(!readSegmentFile(GetConfig(),segPairs)) return;
814 } else {
815 GetConfig().rateParams.entrancePair=runTab->rateEntranceKey->text().toInt();
816 GetConfig().rateParams.exitPair=runTab->rateExitKey->text().toInt();
817 if(GetConfig().rateParams.entrancePair==GetConfig().rateParams.exitPair) {
818 QMessageBox::information(this,tr("No Scattering Rates"),
819 tr("Reaction rates cannot be calculated for elastic scattering."));
820 return;
821 }
822 if(!runTab->fileTempButton->isChecked()) {
823 GetConfig().rateParams.useFile=false;
824 GetConfig().rateParams.minTemp = runTab->minTempText->text().toDouble();
825 GetConfig().rateParams.maxTemp = runTab->maxTempText->text().toDouble();
826 GetConfig().rateParams.tempStep = runTab->tempStepText->text().toDouble();
827 } else {
828 GetConfig().rateParams.useFile=true;
829 GetConfig().rateParams.temperatureFile = runTab->fileTempText->text().toStdString();
830 }
831 SegPairs tempPair = {runTab->rateEntranceKey->text().toInt(),
832 runTab->rateExitKey->text().toInt()};
833 segPairs.push_back(tempPair);
834 }
835 if(segPairs.size()==0) {
836 QMessageBox::information(this,tr("Empty Segments"),tr("No active segments have been found."));
837 return;
838 }
839 int maxPairs=pairsTab->getPairsModel()->getPairs().size();
840 for(std::vector<SegPairs>::const_iterator it = segPairs.begin();it<segPairs.end();it++) {
841 if(it->secondPair==-1) {
842 QList<PairsData> pairsList = pairsTab->getPairsModel()->getPairs();
843 int i;
844 for(i = 0; i<pairsList.size();i++)
845 if(pairsList[i].pairType==10) break;
846 if(i==pairsList.size()) {
847 QMessageBox::information(this,tr("No Capture Pairs"),
848 tr("Total capture is specified, but no capture pairs exist."));
849 return;
850 }
851 } else if(it->firstPair>maxPairs||it->secondPair>maxPairs||it->firstPair<1||it->secondPair<1) {
852 QMessageBox::information(this,tr("Undefined Key"),tr("An undefined pair key is specified."));
853 return;
854 }
855 }
856
857 GetConfig().paramMask &= ~Config::USE_EXTERNAL_CAPTURE;
858 if(!checkExternalCapture(GetConfig(),segPairs)) return;
859 if(GetConfig().paramMask &Config::USE_EXTERNAL_CAPTURE) {
860 if(runTab->oldIntegralsFileButton->isChecked() &&
861 !(GetConfig().paramMask & Config::CALCULATE_REACTION_RATE)) {
862 GetConfig().paramMask |= Config::USE_PREVIOUS_INTEGRALS;
863 GetConfig().integralsfile=runTab->integralsFileText->text().toStdString();
864 } else GetConfig().paramMask &= ~Config::USE_PREVIOUS_INTEGRALS;
865 }
866
867 if(!QDir(QString::fromStdString(GetConfig().outputdir)).exists()) {
868 QMessageBox::information(this,tr("Directory Doesn't Exist"),
869 tr("The specified output directory doesn't exist."));
870 return;
871 }
872 if(!QDir(QString::fromStdString(GetConfig().checkdir)).exists()) {
873 QMessageBox::information(this,tr("Directory Doesn't Exist"),
874 tr("The specified checks directory doesn't exist."));
875 return;
876 }
877 if((GetConfig().paramMask & Config::USE_PREVIOUS_PARAMETERS) &&
878 !QFile(QString::fromStdString(GetConfig().paramfile)).exists()) {
879 QMessageBox::information(this,tr("File Doesn't Exist"),
880 tr("The specified parameter file doesn't exist."));
881 return;
882 }
883 if(((GetConfig().paramMask & Config::USE_PREVIOUS_INTEGRALS) &&
884 (GetConfig().paramMask & Config::USE_EXTERNAL_CAPTURE)) &&
885 !QFile(QString::fromStdString(GetConfig().integralsfile)).exists()) {
886 QMessageBox::information(this,tr("File Doesn't Exist"),
887 tr("The specified integrals file doesn't exist."));
888 return;
889 }
890 if((GetConfig().paramMask & Config::CALCULATE_REACTION_RATE &&
891 GetConfig().rateParams.useFile) &&
892 !QFile(QString::fromStdString(GetConfig().rateParams.temperatureFile)).exists()) {
893 QMessageBox::information(this,tr("File Doesn't Exist"),
894 tr("The specified rate temperature file doesn't exist."));
895 return;
896 }
897
898 if(!(GetConfig().paramMask & Config::CALCULATE_WITH_DATA) &&
899 !(GetConfig().paramMask & Config::CALCULATE_REACTION_RATE)) {
900 QList<TargetIntData> targetIntData = targetIntTab->getTargetIntModel()->getLines();
901 QList<SegmentsTestData> segmentsTestData=segmentsTab->getSegmentsTestModel()->getLines();
902 for(unsigned int i=0;i<targetIntData.size();i++) {
903 if(targetIntData.at(i).isActive==1&&
904 (targetIntData.at(i).isTargetIntegration||targetIntData.at(i).isConvolution)) {
905 unsigned int j=0;
906 unsigned int lastSegNum=0;
907 bool inclusive=false;
908 QList<unsigned int> tempList;
909 QString segmentsList = targetIntData.at(i).segmentsList;
910 while(j<segmentsList.length()) {
911 if(segmentsList[j]>='0'&&segmentsList[j]<='9') {
912 QString tempString;
913 while(segmentsList[j]!=','&&segmentsList[j]!='-'&&
914 j<segmentsList.length()) {
915 tempString+=segmentsList[j];
916 j++;
917 }
918 QTextStream stm(&tempString);
919 unsigned int tempSegNum;stm>>tempSegNum;
920 if(inclusive==true) for(int k=lastSegNum+1;k<=tempSegNum;k++)
921 tempList.push_back(k);
922 else tempList.push_back(tempSegNum);
923 lastSegNum=tempSegNum;
924 }
925 if(segmentsList[j]=='-') inclusive=true;
926 else inclusive =false;
927 j++;
928 }
929 bool isAngularDistribution=false;
930 for(j=0;j<tempList.size();j++) {
931 if(tempList.at(j)<=segmentsTestData.size()) {
932 for(int k = 0; k<segmentsTestData.size(); k++) {
933 if(segmentsTestData.at(k).isActive==1&&
934 tempList.at(j)-1==k&&
935 segmentsTestData.at(k).dataType==3) {
936 isAngularDistribution=true;
937 break;
938 }
939 }
940 }
941 if(isAngularDistribution) break;
942 }
943 if(isAngularDistribution) {
944 QMessageBox::information(this,tr("Incompatable Options"),
945 tr("Angular distribution coefficients cannot be used with convolution or target integration."));
946 return;
947 }
948 }
949 }
950 }
951
952 azureMain = new AZUREMainThread(runTab,GetConfig());
953 connect(azureMain,SIGNAL(finished()),this,SLOT(DeleteThread()));
954 setWindowTitle(QString("AZURE2 -- %1 -- Running").arg(QString::fromStdString(GetConfig().configfile)));
955 runTab->calcButton->setEnabled(false);
956 runTab->stopAZUREButton->setEnabled(true);
957 runTab->runtimeText->SetMouseFiltered(true);
958 startMessage(azureMain->configure());
959 azureMain->start();
960}
