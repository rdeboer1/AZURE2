365QList<ChannelsData> LevelsTab::calculateChannels(int levelIndex) {
366 QList<ChannelsData> channels;
367 QList<PairsData> pairs = pairsModel->getPairs();
368 LevelsData level = (levelsModel->getLevels()).at(levelIndex);
369
370 int maxL=maxLSpin->value();
371 int maxMult=maxMultSpin->value();
372 int maxNumMult=maxNumMultSpin->value();
373
374 for(int i=0;i<pairs.size();i++) {
375 PairsData pair=pairs.at(i);
376 if(pair.pairType==0) {
377 for(double s=fabs(pair.heavyJ-pair.lightJ);s<=pair.heavyJ+pair.lightJ;s+=1.0) {
378 for(double l=fabs(s-level.jValue);l<=s+level.jValue;l+=1.0) {
379 if(int(l*2.0)%2==0&&pair.lightPi*pair.heavyPi*pow(-1,int(l))==level.piValue&&int(l)<=maxL) {
380 ChannelsData channel={0,levelIndex,i,s,int(l),'P',0.0};
381 channels.push_back(channel);
382 }
383 }
384 }
385 } else if(pair.pairType==20) {
386 if(fabs(pair.heavyJ-level.jValue)==0.&&pair.heavyPi==level.piValue) {
387 ChannelsData gtChannel = {0,levelIndex,i,0.,1,'G',0.0};
388 ChannelsData fChannel = {0,levelIndex,i,0.,0,'F',0.0};
389 channels.push_back(gtChannel);
390 channels.push_back(fChannel);
391 } else if(fabs(pair.heavyJ-level.jValue)==1.&&pair.heavyPi==level.piValue) {
392 ChannelsData gtChannel = {0,levelIndex,i,1.,1,'G',0.0};
393 channels.push_back(gtChannel);
394 }
395 } else {
396 int numMult=1;
397 for(int l=1;l<=maxMult;l++) {
398 if(fabs(l-pair.heavyJ)<=level.jValue&&level.jValue<=l+pair.heavyJ&&numMult<=maxNumMult) {
399 QChar radType;
400 int parityChange=pair.heavyPi*level.piValue;
401 if(l%2!=0) {
402 if(parityChange==-1) radType='E';
403 else radType='M';
404 } else {
405 if(parityChange==-1) radType='M';
406 else radType='E';
407 }
408 ChannelsData channel={0,levelIndex,i,pair.heavyJ,l,radType,0.0};
409 channels.push_back(channel);
410 numMult++;
411 }
412 }
413 }
414 }
415 return channels;
416}

584bool LevelsTab::writeNuclearFile(QTextStream& outStream) {

589 outStream.setFieldAlignment(QTextStream::AlignRight);
590
591 double lowJ=0;
592 double highJ=0;
593 for(int la=0;la<levels.size();la++) {
594 double tempJ=levels.at(la).jValue;
595 if(la==0) {
596 lowJ=tempJ;
597 highJ=tempJ;
598 } else if(tempJ<=lowJ) lowJ=tempJ;
599 else if(tempJ>=highJ) highJ=tempJ;
600 }
601 QList<QList<LevelsData> > sortedLevels;
602 QList<QList<int> > levelsMap;
603 for(double j=lowJ;j<=highJ;j+=0.5) {
604 for(int pi=-1;pi<=1;pi+=2) {
605 for(int la=0;la<levels.size();la++) {
606 if(levels.at(la).jValue==j&&levels.at(la).piValue==pi) {
607 if(sortedLevels.size()==0||(sortedLevels.at(sortedLevels.size()-1).at(0).jValue!=j||
608 sortedLevels.at(sortedLevels.size()-1).at(0).piValue!=pi)) {
609 QList<LevelsData> tempLevelList;
610 sortedLevels.append(tempLevelList);
611 QList<int> tempKeyList;
612 levelsMap.append(tempKeyList);
613 }
614 if(sortedLevels.at(sortedLevels.size()-1).size()==0) {
615 sortedLevels[sortedLevels.size()-1].append(levels.at(la));
616 levelsMap[levelsMap.size()-1].append(la);
617 } else for(int mu=0;mu<sortedLevels.at(sortedLevels.size()-1).size();mu++) {
618 if(levels.at(la).energy<=sortedLevels.at(sortedLevels.size()-1).at(mu).energy) {
619 sortedLevels[sortedLevels.size()-1].insert(mu,levels.at(la));
620 levelsMap[levelsMap.size()-1].insert(mu,la);
621 break;
622 }
623 else if(mu==sortedLevels.at(sortedLevels.size()-1).size()-1) {
624 sortedLevels[sortedLevels.size()-1].append(levels.at(la));
625 levelsMap[levelsMap.size()-1].append(la);
626 break;
627 }
628 }
629 }
630 }
631 }
632 }
633 int levelId=1;
634 for(int i=0;i<sortedLevels.size();i++){
635 for(int ii=0;ii<sortedLevels.at(i).size();ii++) {
636 for(int ch=0;ch<channels.size();ch++) {
637 if(channels.at(ch).levelIndex==levelsMap.at(i).at(ii)) {
638 outStream << qSetFieldWidth(4) << sortedLevels.at(i).at(ii).jValue
639 << qSetFieldWidth(5) << sortedLevels.at(i).at(ii).piValue
640 << qSetFieldWidth(13) << sortedLevels.at(i).at(ii).energy
641 << qSetFieldWidth(5) << sortedLevels.at(i).at(ii).isFixed
642 << qSetFieldWidth(5) << "1"
643 << qSetFieldWidth(5) << channels.at(ch).pairIndex+1
644 << qSetFieldWidth(5) << int(channels.at(ch).sValue*2)
645 << qSetFieldWidth(5) << int(channels.at(ch).lValue*2)
646 << qSetFieldWidth(5) << levelId
647 << qSetFieldWidth(5) << sortedLevels.at(i).at(ii).isActive
648 << qSetFieldWidth(5) << channels.at(ch).isFixed
649 << qSetFieldWidth(20) << channels.at(ch).reducedWidth
650 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).lightJ
651 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).lightPi
652 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).heavyJ
653 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).heavyPi
654 << qSetFieldWidth(13) << pairs.at(channels.at(ch).pairIndex).excitationEnergy
655 << qSetFieldWidth(8) << pairs.at(channels.at(ch).pairIndex).lightM
656 << qSetFieldWidth(8) << pairs.at(channels.at(ch).pairIndex).heavyM
657 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).lightZ
658 << qSetFieldWidth(5) << pairs.at(channels.at(ch).pairIndex).heavyZ
659 << qSetFieldWidth(13) << pairs.at(0).seperationEnergy
660 << qSetFieldWidth(13) << pairs.at(channels.at(ch).pairIndex).seperationEnergy
661 << " 0 0 0.0"
662 << qSetFieldWidth(6) << pairs.at(channels.at(ch).pairIndex).pairType
663 << qSetFieldWidth(8) << pairs.at(channels.at(ch).pairIndex).channelRadius
664 << qSetFieldWidth(13) << pairs.at(channels.at(ch).pairIndex).lightG
665 << qSetFieldWidth(13) << pairs.at(channels.at(ch).pairIndex).heavyG
666 << qSetFieldWidth(8) << pairs.at(channels.at(ch).pairIndex).ecMultMask
667 << qSetFieldWidth(0) << endl;
668 }
669 }
670 outStream << endl;
671 levelId++;
672 }
673 }
674
675 outStream.setFieldAlignment(QTextStream::AlignLeft);
676
677 return true;
