183 for(std::map<int,int>::iterator ec=ecPairs.begin();ec!=ecPairs.end();ec++) {
184 PPair *exitPair=this->GetPair(ec->first);
185 if(exitPair->GetPType()!=10) {
186 configure.outStream << "Final state is not a capture pair." << std::endl;

188 }
189 //create new level in compound nucleus for EC state, if it doesn't exist
190 double jValue=exitPair->GetJ(2);
191 int parity=exitPair->GetPi(2);
192 JGroup newJGroup(jValue,parity);
193 int jGroupNum=this->IsJGroup(newJGroup);
194 int levelNum=0;
195 if(jGroupNum) {
196 ALevel newLevel(exitPair->GetExE());
197 levelNum=this->GetJGroup(jGroupNum)->IsLevel(newLevel);

199 this->GetJGroup(jGroupNum)->AddLevel(newLevel);
200 levelNum=this->GetJGroup(jGroupNum)->IsLevel(newLevel);
201 for(int ch=1;ch<=this->GetJGroup(jGroupNum)->NumChannels();ch++) {
202 if(this->GetJGroup(jGroupNum)->GetChannel(ch)->GetRadType()=='P')
203 this->GetJGroup(jGroupNum)->GetLevel(levelNum)->AddGamma(0.1);
204 else this->GetJGroup(jGroupNum)->GetLevel(levelNum)->AddGamma(0.0);
205 }
206 }

560 if(!(configure.paramMask & Config::USE_BRUNE_FORMALISM)) {
561 matrix_r nMatrix;
562 matrix_r mMatrix;
563 for(int mu=0;mu<tempEnergies.size();mu++) {
564 vector_r tempLevelVector;
565 nMatrix.push_back(tempLevelVector);
566 mMatrix.push_back(tempLevelVector);
567 for(int la=0;la<tempEnergies.size();la++) {
568 if(la==mu) {
569 mMatrix[mu].push_back(1.0);
570 double sum=tempEnergies[la];
571 for(int ch=1;ch<=theJGroup->NumChannels();ch++) {
572 if(theJGroup->GetChannel(ch)->GetRadType()=='P')
573 sum+=(shifts[la][ch-1]-theJGroup->GetChannel(ch)->GetBoundaryCondition())*
574 pow(tempGammas[la][ch-1],2.0);
575 }
576 nMatrix[mu].push_back(sum);
577 } else {
578 double mSum=0.0;
579 double nSum=0.0;
580 for(int ch=1;ch<=theJGroup->NumChannels();ch++) {
581 if(theJGroup->GetChannel(ch)->GetRadType()=='P') {
582 mSum+=(shifts[mu][ch-1]-shifts[la][ch-1])/(tempEnergies[mu]-tempEnergies[la])*
583 tempGammas[la][ch-1]*tempGammas[mu][ch-1];
584 nSum+=((tempEnergies[mu]*shifts[la][ch-1]-tempEnergies[la]*shifts[mu][ch-1])/
585 (tempEnergies[mu]-tempEnergies[la])-theJGroup->GetChannel(ch)->GetBoundaryCondition())
586 *tempGammas[la][ch-1]*tempGammas[mu][ch-1];
587 }
588 }
589 mMatrix[mu].push_back(-mSum);
590 nMatrix[mu].push_back(nSum);
591 }
592 }
593 }
594 //solve eigenvalue problem
595 EigenFunc eigenFunc(nMatrix,mMatrix);
596 for(int la=0;la<tempEnergies.size();la++) {
597 theJGroup->GetLevel(levelKeys[la])->SetE(eigenFunc.eigenvalues()[la]);
598 for(int ch=1;ch<=theJGroup->NumChannels();ch++) {
599 double sum=0.0;
600 for(int mu=0;mu<tempEnergies.size();mu++) {
601 sum+=eigenFunc.eigenvectors()[mu][la]*tempGammas[mu][ch-1];
602 }
603 theJGroup->GetLevel(levelKeys[la])->SetGamma(ch,sum);
604 }
605 }
606 } else {
607 for(int la=0;la<tempEnergies.size();la++)
608 for(int ch=1;ch<=theJGroup->NumChannels();ch++)
609 theJGroup->GetLevel(levelKeys[la])->SetGamma(ch,tempGammas[la][ch-1]);
610 }
611 }

1200void CNuc::TransformOut(const Config& configure) {
1201 if(!(configure.paramMask & Config::USE_BRUNE_FORMALISM)) {
1202 int maxIterations=1000;
1203 double energyTolerance=1e-6;
1204 for(int j=1;j<=this->NumJGroups();j++) {
1205 for(int la=1;la<=this->GetJGroup(j)->NumLevels();la++) {
1206 ALevel *theLevel=this->GetJGroup(j)->GetLevel(la);
1207 if(theLevel->IsInRMatrix()) {
1208 int iteration=1;
1209 int thisLevel=0;
1210 bool done=false;
1211 vector_r tempE;
1212 vector_r tempBoundary;
1213 matrix_r tempGamma;
1214 for(int lap=1;lap<=this->GetJGroup(j)->NumLevels();lap++) {
1215 if(this->GetJGroup(j)->GetLevel(lap)->IsInRMatrix()) {
1216 tempE.push_back(this->GetJGroup(j)->GetLevel(lap)->GetFitE());
1217 if(this->GetJGroup(j)->GetLevel(lap)==theLevel) thisLevel=tempE.size()-1;
1218 vector_r tempChanVector;
1219 tempGamma.push_back(tempChanVector);
1220 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1221 tempGamma[tempE.size()-1].push_back(this->GetJGroup(j)->GetLevel(lap)->GetFitGamma(ch));
1222 if(tempE.size()==1) tempBoundary.push_back(this->GetJGroup(j)->GetChannel(ch)->GetBoundaryCondition());
1223 }
1224 }
1225 }
1226 while(iteration<=maxIterations&&!done) {
1227 vector_r boundaryDiff;
1228 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1229 double newBoundary=0.0;
1230 AChannel *theChannel=this->GetJGroup(j)->GetChannel(ch);
1231 PPair *exitPair=this->GetPair(theChannel->GetPairNum());
1232 double localEnergy=tempE[thisLevel]-exitPair->GetSepE()-exitPair->GetExE();
1233 if(theChannel->GetRadType()=='P') {
1234 if(localEnergy<0.0) {
1235 ShftFunc theShiftFunction(exitPair);
1236 newBoundary=theShiftFunction(theChannel->GetL(),tempE[thisLevel]);
1237 }
1238 else {
1239 CoulFunc theCoulombFunction(exitPair,
1240 !!(configure.paramMask&Config::USE_GSL_COULOMB_FUNC));
1241 double radius=exitPair->GetChRad();
1242 newBoundary=theCoulombFunction.PEShift(theChannel->GetL(),radius,localEnergy);
1243 }
1244 boundaryDiff.push_back(newBoundary-tempBoundary[ch-1]);
1245 tempBoundary[ch-1]=newBoundary;
1246 } else boundaryDiff.push_back(boundaryDiff[0]);
1247 }
1248 matrix_r cMatrix;
1249 for(int mu=0;mu<tempE.size();mu++) {
1250 vector_r tempRow;
1251 cMatrix.push_back(tempRow);
1252 for (int mup=0;mup<tempE.size();mup++) {
1253 double chanSum=0.0;
1254 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1255 if(this->GetJGroup(j)->GetChannel(ch)->GetRadType()=='P')
1256 chanSum+=boundaryDiff[ch-1]*tempGamma[mu][ch-1]*
1257 tempGamma[mup][ch-1];
1258 }
1259 if(mu==mup) cMatrix[mu].push_back(tempE[mu]-chanSum);
1260 else cMatrix[mu].push_back(-chanSum);
1261 }
1262 }

1264 if(fabs(eigenFunc.eigenvalues()[thisLevel]-tempE[thisLevel])<=energyTolerance)
1265 done=true;
1266 matrix_r newGamma;
1267 for(int mu=0;mu<tempE.size();mu++) {
1268 vector_r tempChanVector;
1269 newGamma.push_back(tempChanVector);
1270 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1271 double gammaSum=0.0;
1272 for(int mup=0;mup<tempE.size();mup++) {
1273 gammaSum+=eigenFunc.eigenvectors()[mup][mu]*tempGamma[mup][ch-1];
1274 }
1275 newGamma[mu].push_back(gammaSum);
1276 }
1277 }
1278 for(int mu=0;mu<tempE.size();mu++) {
1279 tempE[mu]=eigenFunc.eigenvalues()[mu];
1280 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1281 tempGamma[mu][ch-1]=newGamma[mu][ch-1];
1282 }
1283 }
1284 if(!done) {
1285 if(iteration==maxIterations) {
1286 configure.outStream << "**WARNING: Could Not Transform J = "
1287 << this->GetJGroup(j)->GetJ();
1288 if(this->GetJGroup(j)->GetPi()==-1) configure.outStream << '-';
1289 else configure.outStream << '+';
1290 configure.outStream << " E = " << theLevel->GetFitE() << " MeV**" << std::endl;
1291 tempE[thisLevel]=theLevel->GetFitE();
1292 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++)
1293 tempGamma[thisLevel][ch-1]=theLevel->GetFitGamma(ch);
1294 }
1295 iteration++;
1296 }
1297 }
1298
1299 theLevel->SetTransformE(tempE[thisLevel]);
1300 theLevel->SetTransformIterations(iteration);
1301 double nFSum=1.0;
1302 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1303 AChannel *theChannel=this->GetJGroup(j)->GetChannel(ch);
1304 theLevel->SetTransformGamma(ch,tempGamma[thisLevel][ch-1]);
1305 if(ch<=theLevel->NumNFIntegrals()) nFSum+=2.0*
1306 this->GetPair(theChannel->GetPairNum())->GetChRad()*
1307 this->GetPair(theChannel->GetPairNum())->GetRedMass()*
1308 uconv/pow(hbarc,2.0)*pow(tempGamma[thisLevel][ch-1],2.0)*
1309 theLevel->GetNFIntegral(ch);
1310 }
1311 theLevel->SetSqrtNFFactor(1.0/sqrt(nFSum));
1312 } else {
1313 theLevel->SetTransformE(theLevel->GetFitE());
1314 theLevel->SetTransformIterations(0);
1315 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++)
1316 theLevel->SetTransformGamma(ch,theLevel->GetFitGamma(ch));
1317 }
1318 }
1319 }
1320 } else {
1321 for(int j=1;j<=this->NumJGroups();j++)
1322 for(int la=1;la<=this->GetJGroup(j)->NumLevels();la++) {
1323 this->GetJGroup(j)->GetLevel(la)->SetTransformIterations(0);
1324 this->GetJGroup(j)->GetLevel(la)->SetTransformE(this->GetJGroup(j)->GetLevel(la)->GetFitE());
1325 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++)
1326 this->GetJGroup(j)->GetLevel(la)->
1327 SetTransformGamma(ch,this->GetJGroup(j)->GetLevel(la)->GetFitGamma(ch));
1328 }
1329 }
1330
1331 for(int j=1;j<=this->NumJGroups();j++) {
1332 JGroup *theJGroup=this->GetJGroup(j);
1333 for(int la=1;la<=this->GetJGroup(j)->NumLevels();la++) {
1334 ALevel *theLevel=this->GetJGroup(j)->GetLevel(la);
1335 double normSum=0.0;
1336 vector_r tempPene;
1337 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1338 AChannel *theChannel=this->GetJGroup(j)->GetChannel(ch);
1339 PPair *exitPair=this->GetPair(theChannel->GetPairNum());
1340 double localEnergy=theLevel->GetTransformE()-exitPair->GetSepE()-exitPair->GetExE();
1341 if(theChannel->GetRadType()=='P') {
1342 if(localEnergy<0.0) {
1343 ShftFunc theShiftFunction(exitPair);
1344 normSum+=theShiftFunction.EnergyDerivative(theChannel->GetL(),theLevel->GetTransformE())*
1345 pow(theLevel->GetTransformGamma(ch),2.0);
1346 WhitFunc newWhitFunc(exitPair);
1347 double whitConv=newWhitFunc(theChannel->GetL(),
1348 exitPair->GetChRad(),
1349 fabs(localEnergy));
1350 double pene=exitPair->GetRedMass()*exitPair->GetChRad()*uconv/
1351 pow(hbarc,2.0)/pow(whitConv,2.0);
1352 tempPene.push_back(pene);
1353 }
1354 else {
1355 CoulFunc theCoulombFunction(exitPair,
1356 !!(configure.paramMask&Config::USE_GSL_COULOMB_FUNC));
1357 double radius=exitPair->GetChRad();
1358 normSum+=theCoulombFunction.PEShift_dE(theChannel->GetL(),radius,localEnergy)*
1359 pow(theLevel->GetTransformGamma(ch),2.0);
1360 double pene=theCoulombFunction.Penetrability(theChannel->GetL(),radius,localEnergy);
1361 tempPene.push_back(pene);
1362 }
1363 } else if(theChannel->GetRadType()=='M'||theChannel->GetRadType()=='E') {
1364 if(fabs(theLevel->GetE()-this->GetPair(theChannel->GetPairNum())->GetExE())<1.e-3&&

1369 if(theChannel->GetRadType()=='M'&&theChannel->GetL()==1)
1370 pene=3.0*jValue/4.0/(jValue+1.)/nuclearMagneton/nuclearMagneton;
1371 else if(theChannel->GetRadType()=='E'&&theChannel->GetL()==2)
1372 pene=60.0*jValue*(2.*jValue-1.)/(jValue+1.)/(2.*jValue+3.);
1373 if((int)(2*jValue)%2!=0) pene*=-1.;
1374 tempPene.push_back(pene);
1375 } else {
1376 double pene = (configure.paramMask & Config::USE_RMC_FORMALISM) ? 1.0 : pow(fabs(localEnergy)/hbarc,2.0*theChannel->GetL()+1);
1377 tempPene.push_back(pene);
1378 }
1379 } else tempPene.push_back(1.0);
1380 }
1381 for(int ch=1;ch<=this->GetJGroup(j)->NumChannels();ch++) {
1382 AChannel* theChannel = this->GetJGroup(j)->GetChannel(ch);
1383 complex externalWidth(0.0,0.0);

1385 theLevel->IsInRMatrix()&&(configure.paramMask & Config::USE_EXTERNAL_CAPTURE) &&
1386 !(fabs(theLevel->GetTransformGamma(ch))<1.0e-8 && (configure.paramMask & Config::IGNORE_ZERO_WIDTHS)))
1387 externalWidth=CalcExternalWidth(this->GetJGroup(j),theLevel,
1388 this->GetJGroup(j)->GetChannel(ch),false,configure);
1389 theLevel->SetExternalGamma(ch,externalWidth);
1390 complex totalWidth=theLevel->GetTransformGamma(ch)+externalWidth;
1391 int tempSign = (real(totalWidth)<0.) ? (-1) : (1);
1392 double bigGamma;
1393 if(theChannel->GetRadType()!='F'&&theChannel->GetRadType()!='G')

1398 }
