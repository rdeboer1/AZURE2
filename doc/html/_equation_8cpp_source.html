116void Equation::Parse(const Config &configure) {
117 BuildFunctionList();
118 try{
119 unsigned int position=0;
120 std::vector<TokenPair> stack;
121 //Initial expectation list
122 unsigned char expecting = FUNCTION|NUMBER|PARAMETER|VARIABLE|LEFTPAR;
123 while(position<infixEquation_.length()) {
124 //Retrieve token
125 TokenPair tempPair=GetToken(position,configure);
126 //If last token, enforce additional expectations.
127 if(position>=infixEquation_.length())
128 expecting&=FUNCTION|NUMBER|PARAMETER|VARIABLE|RIGHTPAR;
129 //Check token against expectations
130 if(!(expecting&tempPair.first)) throw SyntaxError(infixEquation_,3,position-1);
131 //Set new expectations for next token
132 if(tempPair.first==NUMBER || tempPair.first==VARIABLE ||
133 tempPair.first==PARAMETER || tempPair.first==FUNCTION) expecting=OPERATOR|RIGHTPAR;
134 else if(tempPair.first==OPERATOR || tempPair.first==LEFTPAR) expecting=FUNCTION|NUMBER|PARAMETER|VARIABLE|LEFTPAR;
135 else if(tempPair.first==RIGHTPAR) expecting=OPERATOR|RIGHTPAR;
136 //Shunting yard algorithm
137 // Numbers, functions, parameters, and variable go directly to output
138 if(tempPair.first==NUMBER||tempPair.first==PARAMETER||tempPair.first==VARIABLE||tempPair.first==FUNCTION) {
139 output_.push_back(tempPair);
140 } else if(tempPair.first==OPERATOR) {
141 if(stack.size()==0) stack.push_back(tempPair);
142 else while(stack.size()>0) {
143 //if top of stack is an operator
144 if(stack[stack.size()-1].first==OPERATOR) {
145 char lastChar=stack[stack.size()-1].second[0];
146 //check precidence and associativity, while conditions are met move from stack to output
147 if((GetOperatorType(tempPair.second[0])<=GetOperatorType(lastChar)&&
148 GetOperatorAssociativity(tempPair.second[0])==LEFT)||
149 (GetOperatorType(tempPair.second[0])<GetOperatorType(lastChar)&&
150 GetOperatorAssociativity(tempPair.second[0])==RIGHT)) {
151 output_.push_back(stack[stack.size()-1]);
152 stack.pop_back();
153 //if stack is empty, push new operator on stack and break loop
154 if(stack.size()==0) {
155 stack.push_back(tempPair);
156 break;
157 }
158 } else {
159 //if conditions aren't met, push new operator on stack and break loop
160 stack.push_back(tempPair);
161 break;
162 }
163 } else {
164 //if top of stack is not operator, push new operator on stack and break loop
165 stack.push_back(tempPair);
166 break;
167 }
168 }
169 } else if(tempPair.first==LEFTPAR) {
170 //left parentheses go directly to the stack
171 stack.push_back(tempPair);
172 } else if(tempPair.first==RIGHTPAR) {
173 //right parentheses initiate push of stack to output until left parenthesis is found
174 while(stack.size()>0&&stack[stack.size()-1].first!=LEFTPAR) {
175 output_.push_back(stack[stack.size()-1]);
176 stack.pop_back();
177 }
178 if(stack.size()==0) {
179 //if stack is empty, parentheses were mismatched
180 throw SyntaxError(infixEquation_,2,position-1);
181 } else if(stack[stack.size()-1].first==LEFTPAR) stack.pop_back();
182 }
183 }
184 //push remaining stack to output after all tokens are read
185 while(stack.size()>0) {
186 //if left parenthesis is found, parentheses were mismatched
187 if(stack[stack.size()-1].first==LEFTPAR) throw SyntaxError(infixEquation_,2);
188 output_.push_back(stack[stack.size()-1]);
189 stack.pop_back();
190 }
191 } catch (SyntaxError e) {
192 configure.outStream << e.what() << std::endl;
193 std::exit(-1);
194 }

259 try {
260 TokenType tempType;
261 std::string tempString;
262 //Number (positive decimal, integer, or exponential notation)
263 if(IsDigit(infixEquation_[position])||infixEquation_[position]=='.') {
264 tempType=NUMBER;
265 while((IsDigit(infixEquation_[position])||infixEquation_[position]=='.'||infixEquation_[position]=='e'||infixEquation_[position]=='E'
266 ||(infixEquation_[position]=='-'&&(infixEquation_[position-1]=='e'||infixEquation_[position-1]=='E'))
267 ||(infixEquation_[position]=='+'&&(infixEquation_[position-1]=='e'||infixEquation_[position-1]=='E')))
268 &&position<infixEquation_.length()) {
269 tempString+=infixEquation_[position];
270 position++;
271 }
272 }
273 //Check for negation. Implimented as a function. MUST be checked before minus is parsed as an operator.
274 else if(infixEquation_[position]=='-'&&(position==0||IsOperator(infixEquation_[position-1])||
275 infixEquation_[position-1]=='(')) {
276 tempType=FUNCTION;
277 position++;
278 tempString="neg";
279 }
280 //Operator (+,-,*,/, or ^)
281 else if(IsOperator(infixEquation_[position])) {
282 tempType=OPERATOR;
283 tempString+=infixEquation_[position];
284 position++;
285 }
286 //Parameter (must be of form a0,a1,...)
287 else if(infixEquation_[position]=='a'&&IsDigit(infixEquation_[position+1])) {
288 tempType=PARAMETER;
289 position++;
290 while(IsDigit(infixEquation_[position])&&position<infixEquation_.length()) {
291 tempString+=infixEquation_[position];
292 position++;
293 }
294 std::istringstream stm;
295 stm.str(tempString);
296 unsigned int paramNumber;stm>>paramNumber;
297 if(paramNumber+1>parameters_.size()) throw SyntaxError(infixEquation_,1,position-1);
298 }
299 //Dependent variable
300 else if(infixEquation_[position]=='x') {
301 tempType=VARIABLE;
302 tempString+=infixEquation_[position];
303 position++;
304 }
305 //Left Parenthesis
306 else if(infixEquation_[position]=='(') {
307 tempType=LEFTPAR;
308 tempString+=infixEquation_[position];
309 position++;
310 }
311 //Right Parenthesis
312 else if(infixEquation_[position]==')') {
313 tempType=RIGHTPAR;
314 tempString+=infixEquation_[position];
315 position++;
316 }
317 //Functions (currently supports whatever is linked in BuildFunctionList())
318 else if(FindFunction(position)) {
319 tempType=FUNCTION;
320 unsigned int length = FindFunction(position);
321 tempString=infixEquation_.substr(position,length);
322 position+=(length+1);
323 }
324 //Unrecognized Tokens
325 else throw SyntaxError(infixEquation_,0,position);
326 //Read subequation if token is function
327 if(tempType==FUNCTION) {
328 std::string subString;
329 //for negation, read until an operator (except ^), or unmatched right paranthesis, is found
330 // and create subequation
331 if(tempString=="neg") {
332 int parenCount=0;
333 while(((infixEquation_[position]!=')'&&

335 !(infixEquation_[position-1]=='e'||infixEquation_[position-1]=='E')))||
336 parenCount!=0)&&
337 position<infixEquation_.length()) {
338 if(infixEquation_[position]=='(') parenCount++;
339 else if(infixEquation_[position]==')') parenCount--;
340 subString+=infixEquation_[position];
341 position++;
342 }
343 if(parenCount>0&&position>=infixEquation_.length()) throw SyntaxError(infixEquation_,2);
344 } else {

346 // and create subequation
347 int parenCount=1;
348 while(parenCount>0&&position<infixEquation_.length()) {
349 if(infixEquation_[position]=='(') parenCount++;
350 else if(infixEquation_[position]==')') parenCount--;
351 if(parenCount!=0) subString+=infixEquation_[position];
352 position++;
353 }
354 if(parenCount>0&&position>=infixEquation_.length()) throw SyntaxError(infixEquation_,2);
355 }
356 Equation subEquation(subString,parameters_,configure);
357 subEquations_.push_back(subEquation);
358 int subIndex=subEquations_.size()-1;
359 std::ostringstream stm;
360 stm << subIndex;
361 tempString+=stm.str();
362 }
363 return TokenPair(tempType,tempString);
364 } catch (SyntaxError e) {

467 double value;

491double Equation::Evaluate(const Config& configure,double x) const {
492 std::vector<TokenPair> localOutput=output_;
493 std::istringstream stm;
494 double result=0.0;
495 int i=0;
496 if(localOutput.size()==1) {
497 result=GetTokenValue(localOutput[0],x,configure);
498 } else {
499 while(localOutput.size()!=1) {
500 if(localOutput[i].first==OPERATOR) {
501 char op=localOutput[i].second[0];
502 double left=GetTokenValue(localOutput[i-2],x,configure);
503 double right=GetTokenValue(localOutput[i-1],x,configure);
504 localOutput.erase(localOutput.begin()+i-2,localOutput.begin()+i+1);
505 localOutput.insert(localOutput.begin()+i-2,
506 TokenPair(NUMBER,BinaryOperation(left,right,op,configure)));
507 i--;
508 }
509 else i++;
510 }
511 stm.clear();
512 stm.str(localOutput[0].second);
513 stm >> result;
514 }
515 return result;
