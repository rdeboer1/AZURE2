188int ESegment::Fill(CNuc *theCNuc, EData *theData, const Config& configure) {
189 std::string infile=this->GetDataFile();
190 std::ifstream in(infile.c_str());
191 if(!in) return -1;
192 while(!in.eof()) {
193 DataLine line(in);
194 if(!in.eof()) {
195 EPoint NewEPoint(line,this);
196 if(this->IsInSegment(NewEPoint)) {
197 this->AddPoint(NewEPoint);
198 PPair *entrancePair=theCNuc->GetPair(theCNuc->GetPairNumFromKey(this->GetEntranceKey()));
199 PPair *exitPair=theCNuc->GetPair(theCNuc->GetPairNumFromKey(this->GetExitKey()));
200 this->GetPoint(this->NumPoints())->SetParentData(theData);
201 if(entrancePair->GetPType()==20) this->GetPoint(this->NumPoints())->ConvertDecayEnergy(exitPair);
202 else this->GetPoint(this->NumPoints())->ConvertLabEnergy(entrancePair);
203 if(exitPair->GetPType()==0&&this->IsDifferential()&&!this->IsPhase()) {
204 if(this->GetEntranceKey()==this->GetExitKey()) {
205 this->GetPoint(this->NumPoints())->ConvertLabAngle(entrancePair);
206 } else {
207 this->GetPoint(this->NumPoints())->ConvertLabAngle(entrancePair,exitPair,configure);
208 }
209 this->GetPoint(this->NumPoints())->ConvertCrossSection(entrancePair,exitPair);
210 }
211 }
212 }
213 }
214 in.close();
215 return 0;
216}
