24{

51bool PlotEntry::readData() {
52 QFile file(filename_);
53 if(!file.open(QIODevice::ReadOnly)) {
54 return false;
55 }
56 QTextStream inStream(&file);
57 QString line("");
58 bool previousLineBreak=false;
59 bool foundBlock=false;
60 int blockNumber=0;
61 while(!inStream.atEnd()&&!foundBlock) {
62 line=inStream.readLine();
63 if(line.trimmed().isEmpty()) {
64 if(!previousLineBreak) {
65 previousLineBreak=true;
66 continue;
67 }
68 if(blockNumber==index_) {
69 foundBlock=true;
70 break;
71 } else {
72 blockNumber++;
73 previousLineBreak=false;
74 points_.clear();
75 continue;
76 }
77 }
78 if(!inStream.atEnd()&&!foundBlock) {
79 QTextStream in(&line);
80 PlotPoint newPoint = {0.,0.,0.,0.,0.,0.,0.,0.,0.};
81 if(type_==0) {
82 in >> newPoint.energy >> newPoint.excitationEnergy >> newPoint.angle >> newPoint.fitCrossSection >> newPoint.fitSFactor
83 >> newPoint.dataCrossSection >> newPoint.dataErrorCrossSection >> newPoint.dataSFactor
84 >> newPoint.dataErrorSFactor;
85 } else {
86 in >> newPoint.energy >> newPoint.excitationEnergy >> newPoint.angle >> newPoint.fitCrossSection >> newPoint.fitSFactor;
87 }
88 points_.push_back(newPoint);
89 }
90 }
91 inStream.flush();
92 file.close();
93 if(!foundBlock) {
94 return false;
95 }
96 hasNegative_=false;
97 for(QVector<PlotPoint>::const_iterator it=points_.begin();
98 it<points_.end();it++) {
99 if(it->fitCrossSection<=0.||
100 (type_==0&&
101 (it->dataCrossSection<=0.||
102 (fabs(it->dataCrossSection)-
103 fabs(it->dataErrorCrossSection))<=0.))) {
104 hasNegative_=true;
105 break;
106 }
107 }
108 return true;
109}
