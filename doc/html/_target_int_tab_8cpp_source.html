304bool TargetIntTab::readFile(QTextStream& inStream) {
305
306 int isActive;
307 QString segmentsList;
308 int numPoints;
309 int isConvolution;
310 double sigma;
311 int isTargetIntegration;
312 double density;
313 QString stoppingPowerEq;
314 int numParameters;
315 QList<double> parameters;
316 int numQCoefficients;
317 QList<double> qCoefficients;
318 int isQCoefficient;
319
320 QString line("");
321 while(!inStream.atEnd()&&line.trimmed()!=QString("</targetInt>")) {
322 line=inStream.readLine();
323 if(line.trimmed().isEmpty()) continue;
324 if(!inStream.atEnd()&&line.trimmed()!=QString("</targetInt>")) {
325 parameters.clear();
326 qCoefficients.clear();
327 QTextStream in(&line);
328 in >> isActive >> segmentsList >> numPoints >> isConvolution >> sigma >> isTargetIntegration
329 >> density >> stoppingPowerEq >> numParameters;
330 if(in.status()!=QTextStream::Ok) return false;
331 int i=0;
332 while(i<numParameters) {
333 double tempParameter;
334 in >> tempParameter;
335 parameters.append(tempParameter);
336 i++;
337 }
338 if(in.status()!=QTextStream::Ok) return false;
339 in >> isQCoefficient;
340 if(in.status()==QTextStream::Ok) {
341 in >> numQCoefficients;
342 i=0;
343 while(i<numQCoefficients) {
344 double tempQCoefficient;
345 in >> tempQCoefficient;
346 qCoefficients.append(tempQCoefficient);
347 i++;
348 }
349 if(in.status()!=QTextStream::Ok) return false;
350 } else {
351 isQCoefficient=0;
352 }
353 bool tempIsQCoefficient=false;
354 if(isQCoefficient==1) tempIsQCoefficient=true;
355 bool tempIsConvolution=false;
356 if(isConvolution==1) tempIsConvolution=true;
357 bool tempIsTargetIntegration=false;
358 if(isTargetIntegration==1) tempIsTargetIntegration=true;
359 TargetIntData newLine = {isActive,segmentsList.remove('\"'),numPoints,tempIsConvolution,sigma,tempIsTargetIntegration,
360 density,stoppingPowerEq.remove('\"'),numParameters,parameters,tempIsQCoefficient,qCoefficients};
361 addLine(newLine);
362 }
363 }
364 targetIntView->resizeRowsToContents();
365 if(line.trimmed()!=QString("</targetInt>")) return false;
366 return true;
